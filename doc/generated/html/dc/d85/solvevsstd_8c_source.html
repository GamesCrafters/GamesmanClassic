41// Data to be stored in each slice of the database
42UINT32 VALUESLOT = 0;
43UINT32 MEXSLOT = 0;
44UINT32 WINBYSLOT = 0;
45UINT32 REMSLOT = 0;
46UINT32 VISITEDSLOT = 0;

48VALUE DetermineValueVSSTD(POSITION position)
49{
50 GMSTATUS status = STATUS_SUCCESS;
51 VALUE gameval = undecided;
52
53 if(!gBitPerfectDB) {
54 status = STATUS_MISSING_DEPENDENT_MODULE;
55 BPDB_TRACE("DetermineValueVSSTD()", "Bit-Perfect DB must be the selected DB to use the slices solver", status);
56 // i wouldn't use this exit call if the status code
57 // was allowed to propogate up
58 exit(0);
59 goto _bailout;
60 }
61
62 //
63 // add slots to database slices
64 //
65
66 status = AddSlot( 2, "VALUE", TRUE, FALSE, FALSE, &VALUESLOT ); // slot 0
67 if(!GMSUCCESS(status)) {
68 BPDB_TRACE("DetermineValueVSSTD()", "Could not add value slot", status);
69 goto _bailout;
70 }
71
72 if(gPutWinBy) {
73 status = AddSlot( 3, "WINBY", TRUE, TRUE, FALSE, &WINBYSLOT ); // slot 2
74 if(!GMSUCCESS(status)) {
75 BPDB_TRACE("DetermineValueVSSTD()", "Could not add winby slot", status);
76 goto _bailout;
77 }
78 }
79
80 if(!kPartizan) {
81 status = AddSlot( 3, "MEX", TRUE, TRUE, FALSE, &MEXSLOT ); // slot 2
82 if(!GMSUCCESS(status)) {
83 BPDB_TRACE("DetermineValueVSSTD()", "Could not add mex slot", status);
84 goto _bailout;
85 }
86 }
87
88 status = AddSlot( 5, "REMOTENESS", TRUE, TRUE, TRUE, &REMSLOT ); // slot 4
89 if(!GMSUCCESS(status)) {
90 BPDB_TRACE("DetermineValueVSSTD()", "Could not add remoteness slot", status);
91 goto _bailout;
92 }
93
94 status = AddSlot( 1, "VISITED", FALSE, FALSE, FALSE, &VISITEDSLOT ); // slot 1
95 if(!GMSUCCESS(status)) {
96 BPDB_TRACE("DetermineValueVSSTD()", "Could not add visited slot", status);
97 goto _bailout;
98 }
99
100 //
101 // allocate
102 //
103
104 status = Allocate();
105 if(!GMSUCCESS(status)) {
106 BPDB_TRACE("DetermineValueVSSTD()", "Could not allocate database", status);
107 goto _bailout;
108 }
109
110 //
111 // use standard dfs solver
112 //
113
114 gameval = DetermineValueVSSTDHelper( position );
115
116_bailout:
117 if(!GMSUCCESS(status)) {
118 return undecided;
119 } else {
120 return gameval;
121 }
122}

124VALUE DetermineValueVSSTDHelper( POSITION position )
125{
126 BOOLEAN foundTie = FALSE, foundLose = FALSE, foundWin = FALSE;
127 MOVELIST *ptr, *head;
128 VALUE value;
129 POSITION child;
130 REMOTENESS maxRemoteness = 0, minRemoteness = MAXINT2;
131 REMOTENESS minTieRemoteness = MAXINT2, remoteness;
132 MEXCALC theMexCalc = 0; /* default to satisfy compiler */
133 int winByValue = 0, minWinByValue = ((1 << (MEX_BITS-1))-1), maxWinByValue = -(1 << (MEX_BITS-1));
134
135 if(GetSlot(position, VISITEDSLOT)) { /* Cycle! */
136 printf("Sorry, but I think this is a loopy game. I give up.");
137 ExitStageRight();
138 exit(0);
139 }
140 /* It's been seen before and value has been determined */
141 else if((value = GetSlot(position, VALUESLOT)) != undecided) {
142 return(value);
143 } else if((value = Primitive(position)) != undecided) {
144 /* first time, end */
145 SetSlot(position, REMSLOT, 0); /* terminal positions have 0 remoteness */
146 if(!kPartizan && !gTwoBits)
147 SetSlot(position, MEXSLOT, MexPrimitive(value)); /* lose=0, win=* */
148 else if (kPartizan && gPutWinBy && !gTwoBits)
149 SetSlot(position, WINBYSLOT, (gPutWinBy(position) & (MEX_MASK >> MEX_SHIFT)));
150 return(SetSlot(position, VALUESLOT, value));
151 /* first time, need to recursively determine value */
152 } else {
153 SetSlot(position, VISITEDSLOT, 1);
154
155 if(!kPartizan && !gTwoBits)
156 theMexCalc = MexCalcInit();
157 head = ptr = GenerateMoves(position);
158 while (ptr != NULL) {
159 MOVE move = ptr->move;
160 gAnalysis.TotalMoves++;
161 child = DoMove(position,ptr->move); /* Create the child */
162
163 if(gSymmetries)
164 child = gCanonicalPosition(child);
165
166 if (child >= gNumberOfPositions)
167 FoundBadPosition(child, position, move);
168
169 value = DetermineValueVSSTDHelper(child); /* DFS call */
170
171 if (kPartizan && gPutWinBy && !gTwoBits) {
172 int childWinByValue = WinByLoad(child);
173 if (childWinByValue < minWinByValue)
174 minWinByValue = childWinByValue;
175 if (childWinByValue > maxWinByValue)
176 maxWinByValue = childWinByValue;
177 }
178
179 if (gGoAgain(position,move))
180 switch(value)
181 {
182 case lose: value=win; break;
183 case win: value=lose; break;
184 default: break; /* value stays the same */
185 }
186
187 //remoteness = Remoteness(child);
188 remoteness = GetSlot(child, REMSLOT);
189 if(!kPartizan && !gTwoBits)
190 theMexCalc = MexAdd(theMexCalc,MexLoad(child));
191 if(value == lose) { /* found a way to give you a lose */
192 foundLose = TRUE; /* thus, it's a winning move */
193 if (remoteness < minRemoteness) minRemoteness = remoteness;
194 }
195 else if(value == tie) { /* found a way to give you a tie */
196 foundTie = TRUE; /* thus, it's a tieing move */
197 if (remoteness < minTieRemoteness) minTieRemoteness = remoteness;
198 }
199 else if(value == win) { /* found a way to give you a win */
200 foundWin = TRUE; /* thus, it's a losing move */
201 if (remoteness > maxRemoteness) maxRemoteness = remoteness;
202 }
203 else
204 BadElse("DetermineValue[1]");
205
206 if (gUseGPS)
207 gUndoMove(move);
208
209 ptr = ptr->next;
210 }
211 FreeMoveList(head);
212 SetSlot(position, VISITEDSLOT, 0);
213
214 if(!kPartizan && !gTwoBits)
215 SetSlot(position, MEXSLOT, MexCompute(theMexCalc));
216 else if (kPartizan && gPutWinBy && !gTwoBits) {
217 int turn = generic_hash_turn(position);
218 if (turn == 1)
219 winByValue = maxWinByValue;
220 else if (turn == 2)
221 winByValue = minWinByValue;
222 else BadElse("Bad generic_hash_turn(position)");
223 SetSlot(position, WINBYSLOT, (winByValue & (MEX_MASK >> MEX_SHIFT)));
224 }
225 if(foundLose) {
226 SetSlot(position, REMSLOT, minRemoteness+1);
227 return (SetSlot(position, VALUESLOT, win));
228 }
229 else if(foundTie) {
230 SetSlot(position, REMSLOT, minTieRemoteness+1);
231 return (SetSlot(position, VALUESLOT, tie));
232 }
233 else if (foundWin) {
234 SetSlot(position, REMSLOT, maxRemoteness+1);
235 return (SetSlot(position, VALUESLOT, lose));
236 }
237 else
238 BadElse("DetermineValue[2]. GenereateMoves most likely didnt return anything.");
239 }
240 BadElse("DetermineValue[3]"); /* This should NEVER be reached */
241 return(undecided); /* But has been added to satisty lint */
242}
