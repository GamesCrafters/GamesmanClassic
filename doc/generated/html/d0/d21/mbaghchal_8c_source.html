812MOVELIST *GenerateMoves(POSITION position) {
813 int goatsLeft, goatsCaptured;
814 char turn;

816 if (goatsCaptured >= goatCaptureGoal) {

821 for (int i = 0; i < boardSize; i++) {
822 if (turn == GOAT && goatsLeft && board[i] == SPACE) {
823 moves = CreateMovelistNode(hashMove(i, i, i), moves);
824 } else if (board[i] == turn && (turn == TIGER || !goatsLeft)) {
825 for (int j = 0; j < numMoves[i]; j++) {
826 int adjIndex = moveMatrix[i][0][j];
827 int jumpIndex = moveMatrix[i][1][j];
828 if (board[adjIndex] == SPACE)
829 moves = CreateMovelistNode(hashMove(i, adjIndex, adjIndex), moves);
830 else if (turn == TIGER && jumpIndex != -1 && board[adjIndex] == GOAT && board[jumpIndex] == SPACE)
831 moves = CreateMovelistNode(hashMove(i, jumpIndex, adjIndex), moves);
832 }
833 }
834 }
835 SafeFree(board);
836 return moves;
837}

1326 numTigers--;
