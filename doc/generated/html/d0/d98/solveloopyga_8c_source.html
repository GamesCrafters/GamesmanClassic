93VALUE lgas_DetermineLoopyValue1(position)
94POSITION position;
95{
96 POSITION child=kBadPosition, parent;
97 MOVE move;
98 POSITIONMOVELIST *ptr;
99 VALUE childValue, adjustedChildValue;
100 REMOTENESS remotenessChild;
101 POSITION i;
102
103 /* Do DFS to set up Parent pointers and initialize KnownList w/Primitives */
104
105
106
107 lgas_SetParents(kBadPosition,position);
108 if(kDebugDetermineValue) {
109 printf("---------------------------------------------------------------\n");
110 printf("Number of Positions = [" POSITION_FORMAT "]\n",gNumberOfPositions);
111 printf("---------------------------------------------------------------\n");
112 // MyPrintParents();
113 printf("---------------------------------------------------------------\n");
114 //MyPrintFR();
115 printf("---------------------------------------------------------------\n");
116 }
117
118 /* Now, the fun part. Starting from the children, work your way back up. */
119 //@@ separate lose/win frontiers
120 while ((gHeadLoseFR != NULL) ||
121 (gHeadWinFR != NULL)) {
122 if ((child = DeQueueLoseFR()) == kBadPosition)
123 child = DeQueueWinFR();
124
125 /* Might as well grab these now, they'll be used later */
126 childValue = GetValueOfPosition(child);
127 remotenessChild = Remoteness(child);
128
129 /* If debugging, print who's in list */
130 if(kDebugDetermineValue)
131 printf("Grabbing " POSITION_FORMAT " (%s) remoteness = %d off of FR\n",
132 child,gValueString[childValue],remotenessChild);
133
134 ptr = lgas_gParents[child];
135
136 /* With losing children, every parent is winning, so we just go through
137 ** all the parents and declare them winning */
138 while (ptr != NULL) {
139
140 parent = ptr->position;
141 move = ptr->move;
142
143 if (parent != kBadPosition && gGoAgain(parent,move)) {
144 switch(childValue) {
145 case win: adjustedChildValue = lose; break;
146 case lose: adjustedChildValue = win; break;
147 default: adjustedChildValue = childValue;
148 }
149 } else {
150 adjustedChildValue = childValue;
151 }
152
153 if (adjustedChildValue == lose) {
154
155 /* Skip if this is the initial position (parent is kBadPosition) */
156 if (parent != kBadPosition) {
157 if (GetValueOfPosition(parent) == undecided) {
158 /* This is the first time we know the parent is a win */
159 InsertWinFR(parent);
160 if(kDebugDetermineValue) printf("Inserting " POSITION_FORMAT " (%s) remoteness = %d into win FR\n",parent,"win",remotenessChild+1);
161 SetRemoteness(parent, remotenessChild + 1);
162 StoreValueOfPosition(parent,win);
163 }
164 else {
165 /* We already know the parent is a winning position. */
166
167 if (GetValueOfPosition(parent) != win) {
168 printf(POSITION_FORMAT " should be win. Instead it is %d.", parent, GetValueOfPosition(parent));
169 BadElse("lgas_DetermineValue");
170 }
171
172 }
173
174
175 } /* while there are still parents */
176
177 /* With winning children */
178 } else if (adjustedChildValue == win) {
179
180 /* Skip if this is the initial position (parent is kBadPosition) */
181 /* If this is the last unknown child and they were all wins, parent is lose */
182 if(parent != kBadPosition && --gNumberChildren[parent] == 0) {
183 /* no more kids, it's not been seen before, assign it as losing, put at head */
184 assert(GetValueOfPosition(parent) == undecided);
185
186 InsertLoseFR(parent);
187 if(kDebugDetermineValue) printf("Inserting " POSITION_FORMAT " (%s) into FR head\n",parent,"lose");
188 /* We always need to change the remoteness because we examine winning node with
189 ** less remoteness first. */
190 SetRemoteness(parent, remotenessChild + 1);
191 StoreValueOfPosition(parent,lose);
192 }
193 } else {
194 BadElse("lgas_DetermineValue found FR member with other than win/lose value");
195 }
196
197 ptr = ptr->next;
198 } /* With children set to other than win/lose. So stop */
199
200 /* We are done with this position and no longer need to keep around its list of parents
201 ** The tie frontier will not need this, either, because this child's value has already
202 ** been determined. It cannot be a tie. */
203 FreePositionMoveList(lgas_gParents[child]);
204 lgas_gParents[child] = NULL;
205
206 } /* while still positions in FR */
207
208 /* Now process the tie frontier */
209
210 while(gHeadTieFR != NULL) {
211 child = DeQueueTieFR();
212 remotenessChild = Remoteness(child);
213
214 ptr = lgas_gParents[child];
215
216 while (ptr != NULL) {
217 parent = ptr->position;
218
219 if(GetValueOfPosition(parent) == undecided) {
220 /* this position has no losing children but has a tieing position so it must be a
221 * tie. Assign its value and set its remoteness. Note that
222 * we give ties with lowest remoteness priority (i.e. if a
223 * position has no losing children, a tieing child of
224 * remoteness 2, and a tieing child of remoteness 10, the
225 * position will be a tie of remoteness 3, not 11. This
226 * decision is pretty arbitrary. We did it this way to be
227 * consistent with DetermineValue for non-loopy games. */
228
229 InsertTieFR(parent);
230 if(kDebugDetermineValue) printf("Inserting " POSITION_FORMAT " (%s) remoteness = %d into win FR\n",parent,"tie",remotenessChild+1);
231 SetRemoteness(parent, remotenessChild + 1);
232 StoreValueOfPosition(parent,tie);
233 }
234 ptr = ptr->next;
235 }
236 FreePositionMoveList(lgas_gParents[child]);
237 lgas_gParents[child] = NULL;
238 }
239
240 /* Now set all remaining positions to tie with remoteness of REMOTENESS_MAX */
241
242 if(kDebugDetermineValue) {
243 printf("---------------------------------------------------------------\n");
244 //MyPrintFR();
245 printf("---------------------------------------------------------------\n");
246 MyPrintParents();
247 printf("---------------------------------------------------------------\n");
248 printf("TIE cleanup\n");
249 }
250
251 for (i = 0; i < gNumberOfPositions; i++)
252 if(Visited(i)) {
253 if(kDebugDetermineValue)
254 printf(POSITION_FORMAT " was visited...",i);
255 if(GetValueOfPosition((POSITION)i) == undecided) {
256 SetRemoteness((POSITION)i,REMOTENESS_MAX);
257 StoreValueOfPosition((POSITION)i,tie);
258 //we are done with this position and no longer need to keep around its list of parents
259 if (lgas_gParents[child]) FreePositionMoveList(lgas_gParents[child]);
260 if(kDebugDetermineValue)
261 printf("and was undecided, setting to tie\n");
262 } else
263 if(kDebugDetermineValue)
264 printf("but was decided, ignoring\n");
265 UnMarkAsVisited((POSITION)i);
266 }
267
268 return(GetValueOfPosition(position));
269}

277void lgas_SetParents (POSITION parent, POSITION root)
278{
279 MOVELIST* moveptr, * movehead;
280 POSITIONLIST* posptr, * thisLevel, * nextLevel;
281 POSITION pos, child;
282 VALUE value;
283 MOVE move = -1; /* initialized to dummy value */
284
285 posptr = thisLevel = nextLevel = NULL;
286 moveptr = movehead = NULL;
287
288 // Check if the top is primitive.
289
290 MarkAsVisited(root);
291 lgas_gParents[root] = CreatePositionMoveNode(parent, move, lgas_gParents[root]);
292 if ((value = Primitive(root)) != undecided) {
293 SetRemoteness(root, 0);
294 switch (value) {
295 case lose: InsertLoseFR(root); break;
296 case win: InsertWinFR(root); break;
297 case tie: InsertTieFR(root); break;
298 default: BadElse("lgas_SetParents found primitive with value other than win/lose/tie");
299 }
300
301 StoreValueOfPosition(root, value);
302 return;
303 }
304
305 thisLevel = StorePositionInList(root, thisLevel);
306
307 while (thisLevel != NULL) {
308 for (posptr = thisLevel; posptr != NULL; posptr = posptr->next) {
309 pos = posptr->position;
310
311 movehead = GenerateMoves(pos);
312
313 for (moveptr = movehead; moveptr != NULL; moveptr = moveptr->next) {
314 move = moveptr->move;
315 child = DoMove(pos, move);
316 ++gNumberChildren[(int)pos];
317 lgas_gParents[(int)child] = CreatePositionMoveNode(pos, move, lgas_gParents[(int)child]);
318
319 if (Visited(child)) continue;
320 MarkAsVisited(child);
321
322 if ((value = Primitive(child)) != undecided) {
323 SetRemoteness(child, 0);
324 switch (value) {
325 case lose: InsertLoseFR(child); break;
326 case win: InsertWinFR(child); break;
327 case tie: InsertTieFR(child); break;
328 default: BadElse("lgas_SetParents found bad primitive value");
329 }
330 StoreValueOfPosition(child, value);
331 } else {
332 nextLevel = StorePositionInList(child, nextLevel);
333 }
334 }
335
336 FreeMoveList(movehead);
337 }
338
339 FreePositionList(thisLevel);
340
341 thisLevel = nextLevel;
342 nextLevel = NULL;
343 }
344}

346POSITIONMOVELIST* CreatePositionMoveNode(POSITION position, MOVE move, POSITIONMOVELIST* next) {
347 POSITIONMOVELIST* result;
348
349 result = (POSITIONMOVELIST*) SafeMalloc(sizeof(POSITIONMOVELIST));
350 result->position = position;
351 result->move = move;
352 result->next = next;
353
354 return result;
355}

358void FreePositionMoveList(POSITIONMOVELIST* posmovelist) {
359 POSITIONMOVELIST* tmp;
360
361 while(posmovelist != NULL) {
362 tmp = posmovelist;
363 posmovelist = posmovelist->next;
