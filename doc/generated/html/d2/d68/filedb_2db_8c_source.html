41static gamesdb_frameid gamesdb_translate(gamesdb* db, gamesdb_pageid vpn) {
42 //the thing it returns must be a frame identified by ppn in the physical buffer
43 gamesdb_frameid ppn = gamesdb_bman_find(db, vpn); //see if it is in physical memory
44
45 if (ppn == NULL) { //the page is not present in physical memory
46 ppn = gamesdb_bman_replace(db, vpn); //get a replacement page, change page table in the process
47
48 if (ppn->valid == GAMESDB_TRUE) {
49 //this page got kicked out by n-chance
50 if (GAMESDB_DEBUG) printf("translate: replaced = %u\n", (unsigned int)ppn);
51 assert(ppn->tag != vpn);
52 if (ppn->tag != vpn) {
53 //buffer page is valid but not the one we want
54 //if (bufp->dirty[bufpage]) //if the page is dirty flush it
55 gamesdb_buf_write(db, ppn);
56 //load in the new page
57 gamesdb_buf_read(db, ppn, vpn);
58 }
59 } else {
60 //load in the new page
61 gamesdb_buf_read(db, ppn, vpn);
62 //set the tag where it is
63 //if (ppn->tag != vpn)
64 //the buffer is uninitialized, this means no record exists in the page
65 }
66 }
67
68 if (GAMESDB_DEBUG) {
69 printf("translate: vpn = %llu, ppn = %u tag = %llu\n", vpn, (unsigned int)ppn, ppn->tag);
70 }
71
72 assert (ppn->tag == vpn);
73 assert (ppn->valid == GAMESDB_TRUE);
74 return ppn;
75}

77/*
78 * allocates memory and sets up a gamescrafters db. Must call destructive
79 * function when done to free up memory.
80 */
81gamesdb* gamesdb_create(int rec_size, gamesdb_pageid max_recs, gamesdb_pageid max_pages, int cluster_size, char* db_name){
82 gamesdb_bman* bman;
83 gamesdb_buffer* bufp;
84 gamesdb_store* storep;
85 gamesdb* data;
86
87 bufp = gamesdb_buf_init(rec_size, max_recs, max_pages);
88 bman = gamesdb_bman_init(bufp);
89
90 data = (gamesdb*) gamesdb_SafeMalloc(sizeof(gamesdb));
91
92 data->buf_man = bman;
93 data->buffer = bufp;
94
95 storep = gamesdb_open(data, db_name, cluster_size);
96 if (storep == NULL) {
97 printf("A fatal error occured when trying to open the database. Aborting.\n");
98 exit(1);
99 }
100 data->store = storep;
101
102 //create initial page
103 gamesdb_buf_addpage(data);
104 bman->clock_hand = bufp->pages;
105
106 return data;
107}

119void gamesdb_get(gamesdb* gdb, char* mem, gamesdb_position pos){
120 gamesdb_buffer* bufp = gdb->buffer;
121
122 gamesdb_pageid vpn = pos / (bufp->buf_size);
123
124 gamesdb_frameid ppn = gamesdb_translate(gdb, vpn);
125
126 //byte offset of the db record (the extra byte + the actual record)
127 gamesdb_offset off = (pos % (bufp->buf_size)) * bufp->rec_size;
128
129 memcpy(mem, ppn->mem+off, bufp->rec_size);
130}
