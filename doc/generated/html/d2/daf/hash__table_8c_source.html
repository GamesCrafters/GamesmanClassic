56/* Move p_entry one up in its list. */
57static void transpose(ght_hash_table_t *p_ht, ght_uint32_t l_bucket, ght_hash_entry_t *p_entry)
58{
59 /*
60 * __ __ __ __
61 * |A_|->|X_|->|Y_|->|B_|
62 * /
63 * => p_entry
64 * __ __/ __ __
65 * |A_|->|Y_|->|X_|->|B_|
66 */
67 if (p_entry->p_prev) /* Otherwise p_entry is already first. */
68 {
69 ght_hash_entry_t *p_x = p_entry->p_prev;
70 ght_hash_entry_t *p_a = p_x ? p_x->p_prev : NULL;
71 ght_hash_entry_t *p_b = p_entry->p_next;
72
73 if (p_a)
74 {
75 p_a->p_next = p_entry;
76 }
77 else /* This element is now placed first */
78 {
79 p_ht->pp_entries[l_bucket] = p_entry;
80 }
81
82 if (p_b)
83 {
84 p_b->p_prev = p_x;
85 }
86 if (p_x)
87 {
88 p_x->p_next = p_entry->p_next;
89 p_x->p_prev = p_entry;
90 }
91 p_entry->p_next = p_x;
92 p_entry->p_prev = p_a;
93 }
94}

96/* Move p_entry first */
97static void move_to_front(ght_hash_table_t *p_ht, ght_uint32_t l_bucket, ght_hash_entry_t *p_entry)
98{
99 /*
100 * __ __ __
101 * |A_|->|B_|->|X_|
102 * /
103 * => p_entry
104 * __/ __ __
105 * |X_|->|A_|->|B_|
106 */
107 if (p_entry == p_ht->pp_entries[l_bucket])
108 {
109 return;
110 }
111
112 /* Link p_entry out of the list. */
113 p_entry->p_prev->p_next = p_entry->p_next;
114 if (p_entry->p_next)
115 {
116 p_entry->p_next->p_prev = p_entry->p_prev;
117 }
118
119 /* Place p_entry first */
120 p_entry->p_next = p_ht->pp_entries[l_bucket];
121 p_entry->p_prev = NULL;
122 p_ht->pp_entries[l_bucket]->p_prev = p_entry;
123 p_ht->pp_entries[l_bucket] = p_entry;
124}

126static void remove_from_chain(ght_hash_table_t *p_ht, ght_uint32_t l_bucket, ght_hash_entry_t *p)
127{
128 if (p->p_prev)
129 {
130 p->p_prev->p_next = p->p_next;
131 }
132 else /* first in list */
133 {
134 p_ht->pp_entries[l_bucket] = p->p_next;
135 }
136 if (p->p_next)
137 {
138 p->p_next->p_prev = p->p_prev;
139 }
140
141 if (p->p_older)
142 {
143 p->p_older->p_newer = p->p_newer;
144 }
145 else /* oldest */
146 {
147 p_ht->p_oldest = p->p_newer;
148 }
149 if (p->p_newer)
150 {
151 p->p_newer->p_older = p->p_older;
152 }
153 else /* newest */
154 {
155 p_ht->p_newest = p->p_older;
156 }
157}

159/* Search for an element in a bucket */
160static ght_hash_entry_t *search_in_bucket(ght_hash_table_t *p_ht, ght_uint32_t l_bucket,
161 ght_hash_key_t *p_key, unsigned char i_heuristics)
162{
163 ght_hash_entry_t *p_e;
164
165 for (p_e = p_ht->pp_entries[l_bucket];
166 p_e;
167 p_e = p_e->p_next)
168 {
169 if ((p_e->key.i_size == p_key->i_size) &&
170 (memcmp(p_e->key.p_key, p_key->p_key, p_e->key.i_size) == 0))
171 {
172 /* Matching entry found - Apply heuristics, if any */
173 switch (i_heuristics)
174 {
175 case GHT_HEURISTICS_MOVE_TO_FRONT:
176 move_to_front(p_ht, l_bucket, p_e);
177 break;
178 case GHT_HEURISTICS_TRANSPOSE:
179 transpose(p_ht, l_bucket, p_e);
180 break;
181 default:
182 break;
183 }
184 return p_e;
185 }
186 }
187 return NULL;
188}

190/* Free a chain of entries (in a bucket) */
191static void free_entry_chain(ght_hash_table_t *p_ht, ght_hash_entry_t *p_entry)
192{
193 ght_hash_entry_t *p_e = p_entry;
194
195 while(p_e)
196 {
197 ght_hash_entry_t *p_e_next = p_e->p_next;
198 he_finalize(p_ht, p_e);
199 p_e = p_e_next;
200 }
201}

213/* Create an hash entry */
214static ght_hash_entry_t *he_create(ght_hash_table_t *p_ht, void *p_data,
215 unsigned int i_key_size, const void *p_key_data)
216{
217 ght_hash_entry_t *p_he;
218
219 /*
220 * An element like the following is allocated:
221 * elem->p_key
222 * / elem->p_key->p_key_data
223 * ____|___/________
224 * |elem|key|key data|
225 * |____|___|________|
226 *
227 * That is, the key and the key data is stored "inline" within the
228 * hash entry.
229 *
230 * This saves space since malloc only is called once and thus avoids
231 * some fragmentation. Thanks to Dru Lemley for this idea.
232 */
233 if ( !(p_he = (ght_hash_entry_t*)p_ht->fn_alloc (sizeof(ght_hash_entry_t)+i_key_size)) )
234 {
235 fprintf(stderr, "fn_alloc failed!\n");
236 return NULL;
237 }
238
239 p_he->p_data = p_data;
240 p_he->p_next = NULL;
241 p_he->p_prev = NULL;
242 p_he->p_older = NULL;
243 p_he->p_newer = NULL;
244
245 /* Create the key */
246 p_he->key.i_size = i_key_size;
247 memcpy(p_he+1, p_key_data, i_key_size);
248 p_he->key.p_key = (void*)(p_he+1);
249
250 return p_he;
251}

253/* Finalize (free) a hash entry */
254static void he_finalize(ght_hash_table_t *p_ht, ght_hash_entry_t *p_he)
255{
256 assert(p_he);
257
258#if !defined(NDEBUG)
259 p_he->p_next = NULL;
260 p_he->p_prev = NULL;
261 p_he->p_older = NULL;
262 p_he->p_newer = NULL;
263#endif /* NDEBUG */
264
265 /* Free the entry */
266 p_ht->fn_free(p_he);
267}

299/* Create a new hash table */
300ght_hash_table_t *ght_create(unsigned int i_size)
301{
302 ght_hash_table_t *p_ht;
303 int i=1;
304
305 if ( !(p_ht = (ght_hash_table_t*)malloc (sizeof(ght_hash_table_t))) )
306 {
307 perror("malloc");
308 return NULL;
309 }
310
311 /* Set the size of the hash table to the nearest 2^i higher then i_size */
312 p_ht->i_size = 1;
313 while(p_ht->i_size < i_size)
314 {
315 p_ht->i_size = 1<<i++;
316 }
317
318 p_ht->i_size_mask = (1<<(i-1))-1; /* Mask to & with */
319 p_ht->i_items = 0;
320
321 p_ht->fn_hash = ght_one_at_a_time_hash;
322
323 /* Standard values for allocations */
324 p_ht->fn_alloc = malloc;
325 p_ht->fn_free = free;
326
327 /* Set flags */
328 p_ht->i_heuristics = GHT_HEURISTICS_NONE;
329 p_ht->i_automatic_rehash = FALSE;
330
331 p_ht->bucket_limit = 0;
332 p_ht->fn_bucket_free = NULL;
333
334 /* Create an empty bucket list. */
335 if ( !(p_ht->pp_entries = (ght_hash_entry_t**)malloc(p_ht->i_size*sizeof(ght_hash_entry_t*))) )
336 {
337 perror("malloc");
338 free(p_ht);
339 return NULL;
340 }
341 memset(p_ht->pp_entries, 0, p_ht->i_size*sizeof(ght_hash_entry_t*));
342
343 /* Initialise the number of entries in each bucket to zero */
344 if ( !(p_ht->p_nr = (int*)malloc(p_ht->i_size*sizeof(int))))
345 {
346 perror("malloc");
347 free(p_ht->pp_entries);
348 free(p_ht);
349 return NULL;
350 }
351 memset(p_ht->p_nr, 0, p_ht->i_size*sizeof(int));
352
353 p_ht->p_oldest = NULL;
354 p_ht->p_newest = NULL;
355
356 return p_ht;
357}

408/* Insert an entry into the hash table */
409int ght_insert(ght_hash_table_t *p_ht,
410 void *p_entry_data,
411 unsigned int i_key_size, const void *p_key_data)
412{
413 ght_hash_entry_t *p_entry;
414 ght_uint32_t l_key;
415 ght_hash_key_t key;
416
417 assert(p_ht);
418
419 hk_fill(&key, i_key_size, p_key_data);
420 l_key = get_hash_value(p_ht, &key) & p_ht->i_size_mask;
421 if (search_in_bucket(p_ht, l_key, &key, 0))
422 {
423 /* Don't insert if the key is already present. */
424 return -1;
425 }
426 if (!(p_entry = he_create(p_ht, p_entry_data,
427 i_key_size, p_key_data)))
428 {
429 return -2;
430 }
431
432 /* Rehash if the number of items inserted is too high. */
433 if (p_ht->i_automatic_rehash && p_ht->i_items > 2*p_ht->i_size)
434 {
435 ght_rehash(p_ht, 2*p_ht->i_size);
436 /* Recalculate l_key after ght_rehash has updated i_size_mask */
437 l_key = get_hash_value(p_ht, &key) & p_ht->i_size_mask;
438 }
439
440 /* Place the entry first in the list. */
441 p_entry->p_next = p_ht->pp_entries[l_key];
442 p_entry->p_prev = NULL;
443 if (p_ht->pp_entries[l_key])
444 {
445 p_ht->pp_entries[l_key]->p_prev = p_entry;
446 }
447 p_ht->pp_entries[l_key] = p_entry;
448
449 /* If this is a limited bucket hash table, potentially remove the last item */
450 if (p_ht->bucket_limit != 0 &&
451 p_ht->p_nr[l_key] >= p_ht->bucket_limit)
452 {
453 ght_hash_entry_t *p;
454
455 /* Loop through entries until the last
456 *
457 * FIXME: Better with a pointer to the last entry
458 */
459 for (p = p_ht->pp_entries[l_key];
460 p->p_next != NULL;
461 p = p->p_next) ;
462
463 assert(p && p->p_next == NULL);
464
465 remove_from_chain(p_ht, l_key, p); /* To allow it to be reinserted in fn_bucket_free */
466 p_ht->fn_bucket_free(p->p_data, p->key.p_key);
467
468 he_finalize(p_ht, p);
469 }
470 else
471 {
472 p_ht->p_nr[l_key]++;
473
474 assert( p_ht->pp_entries[l_key] ? p_ht->pp_entries[l_key]->p_prev == NULL : 1 );
475
476 p_ht->i_items++;
477 }
478
479 if (p_ht->p_oldest == NULL)
480 {
481 p_ht->p_oldest = p_entry;
482 }
483 p_entry->p_older = p_ht->p_newest;
484
485 if (p_ht->p_newest != NULL)
486 {
487 p_ht->p_newest->p_newer = p_entry;
488 }
489
490 p_ht->p_newest = p_entry;
491
492 return 0;
493}

495/* Get an entry from the hash table. The entry is returned, or NULL if it wasn't found */
496void *ght_get(ght_hash_table_t *p_ht,
497 unsigned int i_key_size, const void *p_key_data)
498{
499 ght_hash_entry_t *p_e;
500 ght_hash_key_t key;
501 ght_uint32_t l_key;
502
503 assert(p_ht);
504
505 hk_fill(&key, i_key_size, p_key_data);
506
507 l_key = get_hash_value(p_ht, &key) & p_ht->i_size_mask;
508
509 /* Check that the first element in the list really is the first. */
510 assert( p_ht->pp_entries[l_key] ? p_ht->pp_entries[l_key]->p_prev == NULL : 1 );
511
512 /* LOCK: p_ht->pp_entries[l_key] */
513 p_e = search_in_bucket(p_ht, l_key, &key, p_ht->i_heuristics);
514 /* UNLOCK: p_ht->pp_entries[l_key] */
515
516 return (p_e ? p_e->p_data : NULL);
517}

519/* Replace an entry from the hash table. The entry is returned, or NULL if it wasn't found */
520void *ght_replace(ght_hash_table_t *p_ht,
521 void *p_entry_data,
522 unsigned int i_key_size, const void *p_key_data)
523{
524 ght_hash_entry_t *p_e;
525 ght_hash_key_t key;
526 ght_uint32_t l_key;
527 void *p_old;
528
529 assert(p_ht);
530
531 hk_fill(&key, i_key_size, p_key_data);
532
533 l_key = get_hash_value(p_ht, &key) & p_ht->i_size_mask;
534
535 /* Check that the first element in the list really is the first. */
536 assert( p_ht->pp_entries[l_key] ? p_ht->pp_entries[l_key]->p_prev == NULL : 1 );
537
538 /* LOCK: p_ht->pp_entries[l_key] */
539 p_e = search_in_bucket(p_ht, l_key, &key, p_ht->i_heuristics);
540 /* UNLOCK: p_ht->pp_entries[l_key] */
541
542 if ( !p_e )
543 return NULL;
544
545 p_old = p_e->p_data;
546 p_e->p_data = p_entry_data;
547
548 return p_old;
549}

551/* Remove an entry from the hash table. The removed entry, or NULL, is
552 returned (and NOT free'd). */
553void *ght_remove(ght_hash_table_t *p_ht,
554 unsigned int i_key_size, const void *p_key_data)
555{
556 ght_hash_entry_t *p_out;
557 ght_hash_key_t key;
558 ght_uint32_t l_key;
559 void *p_ret=NULL;
560
561 assert(p_ht);
562
563 hk_fill(&key, i_key_size, p_key_data);
564 l_key = get_hash_value(p_ht, &key) & p_ht->i_size_mask;
565
566 /* Check that the first element really is the first */
567 assert( (p_ht->pp_entries[l_key] ? p_ht->pp_entries[l_key]->p_prev == NULL : 1) );
568
569 /* LOCK: p_ht->pp_entries[l_key] */
570 p_out = search_in_bucket(p_ht, l_key, &key, 0);
571
572 /* Link p_out out of the list. */
573 if (p_out)
574 {
575 remove_from_chain(p_ht, l_key, p_out);
576
577 /* This should ONLY be done for normal items (for now all items) */
578 p_ht->i_items--;
579
580 p_ht->p_nr[l_key]--;
581 /* UNLOCK: p_ht->pp_entries[l_key] */
582#if !defined(NDEBUG)
583 p_out->p_next = NULL;
584 p_out->p_prev = NULL;
585#endif /* NDEBUG */
586
587 p_ret = p_out->p_data;
588 he_finalize(p_ht, p_out);
589 }
590 /* else: UNLOCK: p_ht->pp_entries[l_key] */
591
592 return p_ret;
593}

595static void *first_keysize(ght_hash_table_t *p_ht, ght_iterator_t *p_iterator, const void **pp_key, unsigned int *size)
596{
597 assert(p_ht && p_iterator);
598
599 /* Fill the iterator */
600 p_iterator->p_entry = p_ht->p_oldest;
601
602 if (p_iterator->p_entry)
603 {
604 p_iterator->p_next = p_iterator->p_entry->p_newer;
605 *pp_key = p_iterator->p_entry->key.p_key;
606 if (size != NULL)
607 *size = p_iterator->p_entry->key.i_size;
608
609 return p_iterator->p_entry->p_data;
610 }
611
612 p_iterator->p_next = NULL;
613 *pp_key = NULL;
614 if (size != NULL)
615 *size = 0;
616
617 return NULL;
618}

634static void *next_keysize(ght_hash_table_t *p_ht, ght_iterator_t *p_iterator, const void **pp_key, unsigned int *size)
635{
636 assert(p_ht && p_iterator);
637
638 if (p_iterator->p_next)
639 {
640 /* More entries */
641 p_iterator->p_entry = p_iterator->p_next;
642 p_iterator->p_next = p_iterator->p_next->p_newer;
643
644 *pp_key = p_iterator->p_entry->key.p_key;
645 if (size != NULL)
646 *size = p_iterator->p_entry->key.i_size;
647
648 return p_iterator->p_entry->p_data; /* We know that this is non-NULL */
649 }
650
651 /* Last entry */
652 p_iterator->p_entry = NULL;
653 p_iterator->p_next = NULL;
654
655 *pp_key = NULL;
656 if (size != NULL)
657 *size = 0;
658
659 return NULL;

678void ght_finalize(ght_hash_table_t *p_ht)
679{

681

690 p_ht->pp_entries[i] = NULL;

693 p_ht->pp_entries = NULL;

704/* Rehash the hash table (i.e. change its size and reinsert all
705 * items). This operation is slow and should not be used frequently.
706 */
707void ght_rehash(ght_hash_table_t *p_ht, unsigned int i_size)
708{
709 ght_hash_table_t *p_tmp;
710 ght_iterator_t iterator;
711 const void *p_key;
712 void *p;
713 int i;
714
715 assert(p_ht);
716
717 /* Recreate the hash table with the new size */
718 p_tmp = ght_create(i_size);
719 assert(p_tmp);
720
721 /* Set the flags for the new hash table */
722 ght_set_hash(p_tmp, p_ht->fn_hash);
723 ght_set_alloc(p_tmp, p_ht->fn_alloc, p_ht->fn_free);
724 ght_set_heuristics(p_tmp, GHT_HEURISTICS_NONE);
725 ght_set_rehash(p_tmp, FALSE);
726
727 /* Walk through all elements in the table and insert them into the temporary one. */
728 for (p = ght_first(p_ht, &iterator, &p_key); p; p = ght_next(p_ht, &iterator, &p_key))
729 {
730 assert(iterator.p_entry);
731
732 /* Insert the entry into the new table */
733 if (ght_insert(p_tmp,
734 iterator.p_entry->p_data,
735 iterator.p_entry->key.i_size, iterator.p_entry->key.p_key) < 0)
736 {
737 fprintf(stderr, "hash_table.c ERROR: Out of memory error or entry already in hash table\n"
738 "when rehashing (internal error)\n");
739 }
740 }
741
742 /* Remove the old table... */
743 for (i=0; i<p_ht->i_size; i++)
744 {
745 if (p_ht->pp_entries[i])
746 {
747 /* Delete the entries in the bucket */
748 free_entry_chain (p_ht, p_ht->pp_entries[i]);
749 p_ht->pp_entries[i] = NULL;
750 }
751 }
752
753 free (p_ht->pp_entries);
754 free (p_ht->p_nr);
755
756 /* ... and replace it with the new */
757 p_ht->i_size = p_tmp->i_size;
758 p_ht->i_size_mask = p_tmp->i_size_mask;
759 p_ht->i_items = p_tmp->i_items;
760 p_ht->pp_entries = p_tmp->pp_entries;
761 p_ht->p_nr = p_tmp->p_nr;
762
763 p_ht->p_oldest = p_tmp->p_oldest;
764 p_ht->p_newest = p_tmp->p_newest;
765
766 /* Clean up */
767 p_tmp->pp_entries = NULL;
768 p_tmp->p_nr = NULL;
769 free (p_tmp);
770}
