105 def do_GET(self):
106 unquoted = urllib.unquote(self.path)
107 parsed = urlparse.urlparse(unquoted)
108 self.parsed_path = parsed
109 path = parsed.path.split('/')
110 command = path[-1]
111 if command == 'favicon.ico':
112 return
113 if command == 'getGames':
114 if self.server.status_server is None:
115 self.respond(json.dumps({
116 'status': 'error',
117 'reason': 'Could not get game list.'},
118 indent=Game.indent_response))
119 else:
120 self.respond(json.dumps({
121 'status': 'ok',
122 'response': self.server.status_server.get_table()},
123 indent=Game.indent_response))
124 return
125 game_name = path[-2]
126
127 # Can't use urlparse.parse_qs because of equal signs in board string
128 query = collections.defaultdict(str)
129 query_lst = parsed.query.split('&')
130 for t in query_lst:
131 if t != '':
132 p = t.split('=', 1)
133 if len(p) == 2:
134 query[p[0]] = p[1]
135 else:
136 query[p[0]] = ''
137
138 self.server.log.info('GET: {}'.format(unquoted))
139
140 game = self.server.get_game(game_name)
141
142 # Fix missing quotes on the board string
143 if query['board'] != '' and query['board'][0] != '"':
144 query['board'] = '"{}"'.format(query['board'])
145
146 try:
147 c_command = {
148 'getStart':
149 'start_response',
150
151 'getEnd':
152 'end_response {}'.format(query['board']),
153
154 'getNextMoveValues':
155 'detailed_position_response {}'.format(query['board']),
156
157 'getMoveValue':
158 'move_value_response {}'.format(query['board'])
159 }[command]
160 except KeyError:
161 try:
162 game.respond_to_unknown_request(GameRequest(self,
163 query,
164 command))
165 except Exception:

167 else:

170 def respond(self, response):
171 try:
172 self.server.lock.acquire()
173 self.send_response(200)
174 self.send_header('Content-Length', len(response))
175 self.send_header('Content-Type', 'text/plain')
176 self.send_header('Access-Control-Allow-Origin', '*')
177 self.end_headers()
178 self.wfile.seek(0)
179 headers = self.wfile.read()
180 self.server.log.debug('Sent headers {}.'.format(headers))
181 self.push(headers + response)
182 self.close_when_done()
183 except Exception:
184 pass
185 finally:
186 self.server.lock.release()

250 def close(self):

278 def parse_response(self, response):
279 if 'result' not in response:
280 self.server.log.debug('"result" not in string to parse: {!r}'
281 .format(response))
282 return None
283 lines = response.split('\n')
284 for line in lines:
285 if line.startswith('result'):
286 self.server.log.debug('Parsing line starting with result.')

288 result = line.split('=>>')[1].strip()

291 self.server.log.debug('Parsed json from {!r}'.format(result))
292 except Exception as e:
293 self.server.log.debug('Could not parse: {!r}'.format(result))
294 # Catches problems with split, indexing, and non json
295 # together

299 def handle_timeout(self, request, response):
300 # Did not receive a response from the subprocess, so
301 # exit and kill it.
302 self.server.log.error('Did not receive result '
303 'from {}!'.format(self.game.name))
304 self.server.log.error('Process sent output:\n'
305 '{}'.format(response))
306 if self.process.poll() is not None:
307 # If the process is no longer running
308 self.server.log.error('{} crashed!'.format(self.game.name))
309 request.respond(self.crash_msg)
310 return False
311 else:
312 self.server.log.error('{} timed out!'.format(self.game.name))
313 request.respond(self.timeout_msg)
314 return not close_on_timeout

316 def wait_for_ready(self, timeout):
317 last_output = ''
318 read_timeout = 0.001
319 total_time = read_timeout
320 while total_time < timeout:
321 total_time += read_timeout
322 time.sleep(read_timeout)
323 try:
324 last_output = self.process.stdout.read()
325 except IOError:
326 # Probably no data to read.
327 # Subprocess could have could have crashed, but we'll timeout
328 # anyways.
329 continue
330 self.server.log.debug('{} read:\n'
331 '{}'.format(self.game.name, last_output))
332 if ' ready =>>' in last_output:
333 self.server.log.debug('Found ready prompt in {} seconds.'
334 .format(total_time))
335 return timeout - total_time
336 self.server.log.error('Could not find ready prompt for {}.'
337 .format(self.game.name))
338 return timeout - total_time

340 def handle(self, request):
341 time_remaining = self.wait_for_ready(self.read_timeout)
342 if time_remaining <= 0:
343 return self.handle_timeout(request, '')
344 try:
345 self.server.log.debug('Sent command {}.'.format(request.command))
346 self.process.stdin.write(request.command + '\n')
347 except IOError as e:
348 # This case can be hit if the subprocess crashes between requests
349 # (known to occasionally happen).
350 self.server.log.error('{} crashed on write!'
351 .format(self.game.name))
352 request.respond(self.crash_msg)
353 return False
354 response = ''
355 timeout = 0.01
356 parsed = None
357 curr_time = time.time()
358 while not parsed:
359 if time_remaining <= 0:
360 self.server.log.debug('timeout')
361 return self.handle_timeout(request, response)
362 time_remaining -= timeout
363 time.sleep(timeout)
364 try:
365 next_char = ''
366 while next_char != '\n':
367 response += next_char
368 next_char = self.process.stdout.read(1)
369 if time.time() - curr_time > 20:
370 #if time.time() - curr_time > 5000:
371 self.server.log.debug('timeout')
372 return self.handle_timeout(request, response)
373 except IOError:
374 continue
375 parsed = self.parse_response(response)
376 request.respond(parsed)
377 return True

379 def request_loop(self):
380 live = True
381 total_idle_time = 0.0
382 while live:
383 self.server.log.debug(
384 'In request loop for {}.'.format(self.game.name))
385 try:
386 request = self.queue.get(block=True,
387 timeout=self.req_timeout_step)
388 except Queue.Empty as e:
389 total_idle_time += self.req_timeout_step
390 if total_idle_time > self.req_timeout:
391 self.server.log.error(
392 '{} closed from lack of use.'.format(self.game.name))
393 live = False
394 else:
395 total_idle_time = 0.0
396 live = self.handle(request)
397 self.server.log.debug('{} is using {}% of memory.'
398 .format(self.game.name,
399 self.memory_percent_usage()))
400 if self.memory_percent_usage() > max_memory_percent_per_process:
401 self.server.log.error('{} used too much memory!'
402 .format(self.game.name))
403 live = False
404 self.close()
