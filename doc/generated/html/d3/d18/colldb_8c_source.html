35/* this defines the maximum number of bits that can be in the hash field
36 * of a position.
37 * (thus limiting the maximum length of the hash array to be (2^HASH_FIELD_MAX)*40 bits)
38 * the maximum size of the db should be
39 * (2^HASH_FIELD_MAX)*32 + (2^HASH_FIELD_MAX + 8)*40
40 * the first term needs to be continuous space, the second one consists of sparse 40-bit spaces
41 * ----------- hand tuning with big games possible -------------
42 */
43#define HASH_FIELD_MAX 16

65/* the concept here is to divide the POSITION into two fields */
66
67/* 00000000000000000000000000000000
68 * | key | hash index |
69 * The key is stored in the chain hanging off the bucket
70 * the hash index determines which bucket the POSITION goes in.
71 * the database always try to minimize the hash index for maximum capacity
72 * due to memory restrictions the hash index may reach its theoretical
73 * maximum around 2^20 (because the array needs continuous space)
74 * so this db may not make do with all 64 bits of the POSITION
75 * but it is better than the array db, since it has more tolerance
76 * for fragmentation
77 */
78

106/* this guarantees that in the case of a tight hash,
107 * and all POSITIONs are valid,
108 * each bucket contains as many POSITIONs as possible
109 * (all 2^8 keys allowed are utilized)
110 */
111unsigned int calc_field_len(POSITION total)
112{
113 unsigned int shamt = 0;
114
115 while (total > ((1<<8)-1)) {
116 total = total >> 1;
117 shamt++;
118 }
119
120 if (shamt > HASH_FIELD_MAX) {
121 /* the hash index field exceeds a predefined value */
122 printf("this is probably too big to be handled by the collision database. I give up.\n");
123 ExitStageRight();
124 exit(0);
125 }
126
127 return shamt;
128}

131void colldb_init(DB_Table *new_db)
132{
133 hitcount = 0;
134 hash_field_len = calc_field_len(gNumberOfPositions);
135 hash_mask = (1 << hash_field_len) - 1;
136/*
137 * POSITION temp = hash_mask;
138 *
139 * printf("hash_mask = ");
140
141 * while(temp > 1) {
142 * printf("%d", (int)(temp & 1));
143 * temp = temp >> 1;
144 * }
145
146 * printf("\n");
147 */
148 //setup internal memory table
149 colldb_hash_table = (colldb_value_node **) SafeMalloc ((1 << hash_field_len) * sizeof(colldb_value_node*));
150 //printf("initalizing collision database. %d rows allocated.\n",colldb_HASHSIZE);
151
152 memset(colldb_hash_table, 0x0000, (1 << hash_field_len) * sizeof(colldb_value_node*));
153
154 //set function pointers
155 new_db->get_value = colldb_get_value;
156 new_db->put_value = colldb_set_value;
157 new_db->get_remoteness = colldb_get_remoteness;
158 new_db->put_remoteness = colldb_set_remoteness;
159 new_db->check_visited = colldb_check_visited;
160 new_db->mark_visited = colldb_mark_visited;
161 new_db->unmark_visited = colldb_unmark_visited;
162 new_db->get_mex = colldb_get_mex;
163 new_db->put_mex = colldb_set_mex;
164
165 new_db->free_db = colldb_free;
166
167 return;
168}

170void colldb_free(){
171
172 int i;
173 colldb_value_node *next, *cur;
174
175 if(colldb_hash_table) {
176 for(i=0; i<(1 << hash_field_len); i++) {
177 cur = colldb_hash_table[i];
178 while(cur != NULL) {
179 next = cur->next;
180 SafeFree(cur);
181 cur = next;
182 }
183 }
184 SafeFree(colldb_hash_table);
185 }
186}

188colldb_value_node *colldb_find_pos(POSITION position){
189
190 POSITION i;
191 char key;
192 colldb_value_node *cur;
193
194 key = (char) (position >> hash_field_len);
195 i = position & hash_mask;
196 cur = colldb_hash_table[i];
197
198 while(cur != NULL) {
199 if(cur->myPos == key) {
200 /*
201 * hitcount++;
202 * if(hitcount%1000 == 0){
203 * printf("hit: %d, access: %d, visit: %d, remote: %d\n",hitcount,accesscount,visitcount,remotecount);
204 *}
205 */
206 return cur;
207 }
208 cur = cur->next;
209 }
210
211 return NULL;
212}

240VALUE colldb_set_value(POSITION pos, VALUE val)
241{
242 colldb_value_node *ptr = colldb_find_pos(pos);
243
244 if(ptr == NULL)
245 ptr = colldb_add_node(pos);
