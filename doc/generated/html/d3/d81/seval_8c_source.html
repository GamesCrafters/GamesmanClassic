102void chooseEvaluator(int numInEvaluatorList){
103 seList temp = evaluatorList;
104 while( numInEvaluatorList-- > 0 && temp!=NULL) temp = temp->next;
105
106 if(temp!=NULL) {
107 if(currEvaluator!=NULL)
108 freeEvaluatorList(currEvaluator);
109 currEvaluator = SafeMalloc(sizeof(struct seNode));
110 memcpy(currEvaluator, temp, sizeof(struct seNode));
111 currEvaluator->name = copyString(temp->name);
112 currEvaluator->featureList = copyFeatureList(temp->featureList);
113 currEvaluator->next = NULL;
114 printf("\nThe Evaluator has been set to %s\n", currEvaluator->name);
115 gSEvalLoaded = TRUE;
116 }
117}

137void updateCurrentEvaluator(){
138 seList srcEvaluator = findEvaluator(currEvaluator->name);
139 if(srcEvaluator==NULL) {
140 freeEvaluatorList(currEvaluator);
141 gSEvalLoaded = FALSE;
142 }
143 else{
144 currEvaluator->variant = srcEvaluator->variant;
145 currEvaluator->perfect = srcEvaluator->perfect;
146 currEvaluator->next = NULL;
147 if(currEvaluator->featureList!=NULL)
148 freeFeatureList(currEvaluator->featureList);
149 currEvaluator->featureList = copyFeatureList(srcEvaluator->featureList);
150 }

176fList parseFeature(scew_element* sEvalNode){
177 int i;
178 float tempEvalParams[4];
179 scew_element* element = NULL;
180 fList tempList = NULL;
181 while ((element = scew_element_next(sEvalNode, element)) != NULL) {
182 fList currFeature = SafeMalloc(sizeof(struct fNode));
183 scew_attribute* attribute = NULL;
184
185
186 // Default values for feature:
187 currFeature->scale = &linear;
188 currFeature->weight = 1;
189
194 attribute = NULL;
195 while ((attribute = scew_attribute_next(element, attribute)) != NULL)
196 {
197 if(strcmp(scew_attribute_name(attribute),"name")==0)
198 currFeature->name = copyString((char *)scew_attribute_value(attribute));
199 else if(strcmp(scew_attribute_name(attribute),"type")==0)
200 currFeature->type = strcmp((char *)scew_attribute_value(attribute),"custom")==0 ? custom : library;
201 else if(strcmp(scew_attribute_name(attribute),"scaling")==0) {
202 if(strcmp(scew_attribute_value(attribute),"linear")==0) {
203 currFeature->scale = &linear;
204 } else if(strcmp(scew_attribute_value(attribute),"logistic")==0) {
205 currFeature->scale = &logistic;
206 } else if(strcmp(scew_attribute_value(attribute),"quadratic")==0) {
207 currFeature->scale = &quadratic;
208 } else if(strcmp(scew_attribute_value(attribute),"logarithmic")==0) {
209 currFeature->scale = &logarithmic;
210 } else if(strcmp(scew_attribute_value(attribute),"boolean")==0) {
211 currFeature->scale = &boolean;
212 }
213 }
214 else if(strcmp(scew_attribute_name(attribute),"functionParameters")==0) {
215 setArrayFromString(tempEvalParams, (char *)scew_attribute_value(attribute));
216 for(i=0; i<SEVAL_NUMEVALPARAMS; i++) {
217 currFeature->evalParams[i] = (int) tempEvalParams[i];
218 }
219 }
220 else if(strcmp(scew_attribute_name(attribute),"scalingParams")==0)
221 setArrayFromString(currFeature->scaleParams, (char *)scew_attribute_value(attribute));
222 else if(strcmp(scew_attribute_name(attribute),"piece")==0)
223 currFeature->piece = atoi((char *)scew_attribute_value(attribute));
224 else if(strcmp(scew_attribute_name(attribute),"weight")==0)
225 currFeature->weight = atof((char *)scew_attribute_value(attribute));
226 else {
227 BadElse("parse_element");
228 printf("Bad attribute=\"%s\"",(char *)scew_attribute_value(attribute));
229 }
230 }
231
232 currFeature->fEvalC = NULL;
233 currFeature->fEvalL = NULL;
234
235 // Now we should have the name
236 if( currFeature->type == custom && gGetSEvalCustomFnPtr!=NULL)
237 currFeature->fEvalC = (featureEvaluatorCustom)gGetSEvalCustomFnPtr(currFeature->name);
238 else
239 currFeature->fEvalL = (featureEvaluatorLibrary)getSEvalLibFnPtr(currFeature->name);
240
241 if((currFeature->fEvalC == NULL) && (currFeature->fEvalL == NULL)) {
242 printf("A function for feature \"%s\" could not be found!\n",currFeature->name);
243 return NULL;
244 }
245
246 currFeature->next = tempList;
247 tempList = currFeature;
248 }
249
250 return tempList;
251}

266seList parseEvaluators(scew_element* element, seList tempList) {
267 scew_element* child = NULL;
268 seList currEvaluator = NULL;
269
270 if (element == NULL)
271 {
272 return NULL;
273 }
274
275 if(strcmp(scew_element_name(element),"seval")==0) {
276 currEvaluator = SafeMalloc(sizeof(struct seNode));
277 scew_attribute* attribute = NULL;
278 currEvaluator->name = NULL;
279 //currEvaluator->element = element;
280 currEvaluator->variant = -1;
281 currEvaluator->perfect = FALSE;
282 while ((attribute = scew_attribute_next(element, attribute)) != NULL)
283 {
284 if(strcmp(scew_attribute_name(attribute),"name")==0)
285 currEvaluator->name = copyString((char *)scew_attribute_value(attribute));
286 else if(strcmp(scew_attribute_name(attribute),"variant")==0)
287 currEvaluator->variant = atoi((char *)scew_attribute_value(attribute));
288 else if(strcmp(scew_attribute_name(attribute),"perfect")==0) {
289 currEvaluator->perfect = (strcmp(scew_attribute_value(attribute),"true")==0) ? TRUE : FALSE;
290 if(currEvaluator->perfect) gSEvalPerfect = TRUE;
291 } else{
292 BadElse("parseEvaluators");
293 printf("Bad attribute=\"%s\"",(char *)scew_attribute_value(attribute));
294 }
295 }
296
297 if( currEvaluator->name==NULL )
298 currEvaluator->name = copyString("default");
299
300 currEvaluator->featureList = parseFeature(element);
301
302 currEvaluator->next = tempList;
303 tempList = currEvaluator;
304 }
305
311 child = NULL;
312 while ((child = scew_element_next(element, child)) != NULL)
313 {
314 tempList = parseEvaluators(child, tempList);
315 }
316
317 return tempList;
318}

320scew_element* createEvaluatorNode(seList evaluator) {
321 char placeHolderString[30];
322 STRING scaleFnName=NULL;
323 fList temp = evaluator->featureList;
324
325 memset(placeHolderString,0,30);
326
327 scew_element* evaluatorNode = scew_element_create("seval");
328 scew_element_add_attr_pair(evaluatorNode, "name", evaluator->name);
329 sprintf(placeHolderString,"%d", evaluator->variant);
330 scew_element_add_attr_pair(evaluatorNode, "variant", placeHolderString);
331
332 while ( temp!=NULL ) {
333 scew_element* featureNode = scew_element_create("feature");
334
335 scew_element_add_attr_pair(featureNode, "name", temp->name);
336
337 sprintf(placeHolderString,"%.3f", temp->weight);
338 scew_element_add_attr_pair(featureNode, "weight", placeHolderString);
339
340 sprintf(placeHolderString,"%d", temp->piece);
341 scew_element_add_attr_pair(featureNode, "piece", placeHolderString);
342
343 sprintf(placeHolderString,"%.3f, %.3f, %.3f, %.3f", (double)temp->evalParams[0], (double)temp->evalParams[1], (double)temp->evalParams[2], (double)temp->evalParams[3]);
344 scew_element_add_attr_pair(featureNode, "functionParameters", placeHolderString);
345
346 scew_element_add_attr_pair(featureNode, "type", (temp->type==custom) ? "custom" : "library");
347
348 scaleFnName = getScaleFnName(temp);
349 scew_element_add_attr_pair(featureNode, "scaling", scaleFnName);
350
351 sprintf(placeHolderString,"%.3f, %.3f, %.3f, %.3f", (double)temp->scaleParams[0], (double)temp->scaleParams[1], (double)temp->scaleParams[2], (double)temp->scaleParams[3]);
352 scew_element_add_attr_pair(featureNode, "scalingParams", placeHolderString);
353
354 scew_element_add_elem(evaluatorNode, featureNode);
355
356 temp = temp->next;
357 }
358 return evaluatorNode;
359}

379void TryToLoadAnEvaluator(){
380 char fileName[30];
381 sprintf(fileName, "xml/%s.xml", kDBName);
382 if(!FullInitializeSEval(fileName)) return;
383
384 // If we don't have an evaluator defined, then
385 // loop through them and pick one
386 if(currEvaluator == NULL && evaluatorExistsForVariant(getOption())) {
387 seList bestEvaluator = NULL;
388 int bestEvalNum = -1, traverserNum=0;
389 seList traverser = evaluatorList;
390
391 while (traverser != NULL) {
392 if( (traverser->variant==-1 || traverser->variant==getOption()) ) {
393 if(bestEvaluator!=NULL) {
394 if( bestEvaluator->perfect!=TRUE && traverser->perfect==TRUE) {
395 bestEvaluator = traverser;
396 bestEvalNum = traverserNum;
397 }
398 }
399 else{
400 bestEvaluator=traverser;
401 bestEvalNum = traverserNum;
402 }
403 }
404
405 if(bestEvaluator->perfect == TRUE) {
406 // Set currEvaluator and break
407 chooseEvaluator(bestEvalNum);
408 return;
409 }
410 traverser = traverser->next;
411 traverserNum++;
412 }
413
414 if(bestEvalNum!=-1)
415 chooseEvaluator(bestEvalNum);
416 }
417
418 return;
419}

598float evaluatePosition(POSITION p){
599 fList features = NULL;
600 float valueSum = 0;
601 float weightSum = 0;
602 void* board = NULL;
603
604 if(currEvaluator!=NULL)
605 features = currEvaluator->featureList;
606 else{
607 BadElse("evaluatePosition");
608 printf("Tried to call evaluatePosition when currEvaluator==NULL");
609 }
610
611
612 if(linearUnhash != NULL)
613 board = (*linearUnhash)(p);
614
615 while(features!=NULL) {
616 if(features->type == library) {
617 if( linearUnhash!=NULL )
618 valueSum += features->weight * // += weight * scale(evaluateTrait(board))
619 features->scale(features->fEvalL(board,(features->piece==initial ? lBoard.initialPlayerPiece : lBoard.opponentPlayerPiece)
620 ,features->evalParams),features->scaleParams);
621 else{
622 BadElse("evaluatePosition");
623 printf("linearUnhash MUST be set if library functions are used!\n");
624 printf("Results will be nondeterministic.\n");
625 return -2;
626 }
627 } else if(features->type == custom) {
628 valueSum += features->weight * features->scale(features->fEvalC(p),features->scaleParams);
629 } else {
630 printf("ERROR, unknown feature type");
631 }
632
633 weightSum += features->weight;
634 features = features->next;
635 }
636
637 if(board!=NULL)
638 SafeFree(board);
639
640 return (valueSum/weightSum);
641}

648//Params are {-1 intersect, 1 intersect}
649float linear(float value,float params[]){
650 //return (value < params[0]) ? -1 : ((value > params[1]) ? 1 : value);
651 return value;
652}
653
654//Params are {shift,time scale} of sigmoid function
655float logistic(float value,float params[]){
656 float ans = (1/(1+exp(-params[1]*(value-params[0]))));
657 return ans;
658}
659
660//Params are {isConcaveDown,-1 intersect,1 intersect}
661float quadratic(float value,float params[]){
662 float ans;
663 float width = params[2] - params[1];
664 if(params[0]==0) {
665 ans = (2/(width*width))*((value - params[1]) * (value - params[1]));
666 } else {
667 ans = 1 - (2/(width*width))*((params[2] - value) * (params[2] - value));
668 }
669
670 ans = (value < params[1]) ? -1 : ((value > params[2]) ? 1 : ans);
671
672 return ans;
673}
674
675#ifdef WIN32
676float booleanA(float value,float params[]) {
677 return (value>params[0] ? 1 : -1);
678}
679#else
680float boolean(float value,float params[]) {
681 return (value>params[0] ? 1 : -1);
682}
683#endif
684
685
686//Params are {value*10 where y==1 for log10(abs(value*9)+1) (Default=1)}
687float logarithmic(float value,float params[]){
688 float absValue = fabs(value);
689 int sign = (int) value / absValue;
690 float scaleFactor = (params==NULL || params[0]==0) ? 1 : params[0];
691 float ans = log10( (absValue*9+1) / scaleFactor );
692 return (sign * (absValue>1 || ans>1)) ? 1 : ans;
693}

974 **the board. 0 = top, 1 = right, 2 = bottom, 3 = left.

976float numFromEdge(void* board,void* piece,int params[]) {
977 int count=0,i;

979

998/*Returns a measurement of the clustering of pieces on the board. First
999 **calculates the centroid, then measures the mean squared distance of all
1000 **pieces from the centroid.
1001 */
1002float clustering(void* board,void* piece,int params[]) {
1003 int rowAverage=0,colAverage=0,numPieces=0,i=0;
1004 float value=0;
1005 for(i=0; i<lBoard.size; i++) {
1006 if(compare(i,piece,board)) {
1007 rowAverage+=getRow(i);
1008 colAverage+=getCol(i);
1009 numPieces++;
1010 }
1011 }
1012 if(numPieces==0) {
1013 return 0;
1014 }
1015 //compute actual averages
1016 rowAverage = rowAverage/numPieces;
1017 colAverage = colAverage/numPieces;
1018
1019 for(i=0; i<lBoard.size; i++) {
1020 if(compare(i,piece,board)) {
1021 value += (getRow(i)-rowAverage)*(getRow(i)-rowAverage) + (getCol(i)-colAverage)*(getCol(i)-colAverage);
1022 }
1023 }
1024 value /= numPieces;
1025 return value;

1029 **A connection is defined as a string of blank spaces, possibly
1030 **diagonal, between two of the given piece.
1031 */
1032float connections(void* board,void* piece,int params[]) {
1033 int count=0,i,j,pathLength,loc,direction;
1034 BOOLEAN diagonals = params[0];
1035 void* blank = lBoard.blankPiece;
1036 for(i=0; i<lBoard.size; i++) {
1037 if (compare(i,piece,board)) {
1038 for(direction = 3; direction<7; direction+= (diagonals ? 1 : 2)) {
1039
1040 //compute distance to edge of board along direction
1041 if (direction==3) {
1042 pathLength = lBoard.cols-1-getCol(i);
1043 } else if (direction==4) {
1044 pathLength = min(lBoard.cols-1-getCol(i),lBoard.rows-1-getRow(i));
1045 } else if (direction==5) {
1046 pathLength = lBoard.rows-1-getRow(i);
1047 } else { //direction==6
1048 pathLength = min(lBoard.rows-1-getRow(i),getCol(i));
1049 }
1050
1051 loc =i;
1052 for(j=0; j<pathLength; j++) {
1053 loc+=lBoard.directionMap[direction];
1054 if (compare(loc,piece,board)) {
1055 count++;
1056 break;
1057 } else if (!compare(loc,blank,board)) {
1058 break;
1059 }
1060 }

1062 }
1063 }
1064 }
