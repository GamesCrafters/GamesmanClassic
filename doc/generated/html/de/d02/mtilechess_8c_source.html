275void InitializeGame ()
276{
277 int i, counter = 0, boardSize = (int)(pow((int)sqrt(strlen(theBoard))-2,2)), totalPieces = (int)sqrt(boardSize);
278 long long unsigned largestBoard = 0;
279 char *pieces = SafeMalloc(totalPieces+1);
280
281 kUsePureDraw = TRUE; /* This game is known to be a pure draw game. */
282
283 for (i = 0; i < strlen(theBoard); i++) {
284 if (theBoard[i] != ' ') {
285 pieces[counter++] = theBoard[i];
286 }
287 }
288 pieces[counter] = '\0';
289 gMoveToStringFunPtr = &MToS;
290 for (i = 0; i < boardSize; i += (totalPieces+1)) {
291 largestBoard = largestBoard | (long long unsigned)pow(2,i);
292 }
293 boardToIndex = (int*)SafeMalloc(sizeof(int)*pow(2,totalPieces*totalPieces));
294 indexToBoard = (int*)SafeMalloc(sizeof(int)*pow(2,totalPieces*totalPieces));
295 PreProcess(pieces); // Generate the legal placements for pieces
296 /*gNumberOfPositions = (((MaxPlacementIndex*MaxPiecePermutationValue) +
297 MaxPiecePermutationValue-1) << 1) + MaxPlayerTurnValue; */
298 gNumberOfPositions = (boardToIndex[largestBoard]*BMAX) + BMAX;
299 /*
300 gInitialPosition = (boardToIndex[7]*BMAX) + generic_hash_hash(pieces,theCurrentPlayer)+0;//+ 0 for offset
301 */
302 gInitialPosition = hashBoard(theBoard,theCurrentPlayer);
303 SafeFree(pieces);
304 SetupTierStuff();
305}

307/************************************************************************
308**
309** NAME: GenerateMoves
310**
311** DESCRIPTION: Creates a linked list of every move that can be reached
312** from this position. Returns a pointer to the head of the
313** linked list.
314**
315** INPUTS: POSITION position : Current position for move
316** generation.
317**
318** OUTPUTS: (MOVELIST *) : A pointer to the first item of
319** the linked list of generated moves
320**
321** CALLS: MOVELIST *CreateMovelistNode();
322**
323************************************************************************/
324
325MOVELIST *GenerateMoves (POSITION position)
326{
327 MOVELIST *moves = NULL;
328 int currentPlayer, i, length;
329 char piece;
330 char *boardArray;
331 boardArray = unhashBoard(position);
332 currentPlayer = getCurrTurn(position);
333 length = strlen(boardArray);
334 for (i = 0; i < length; i++) {
335 piece = boardArray[i];
336 if (isSameTeam(piece,currentPlayer)) {
337 // The piece we're look at is the current player's piece
338 if (piece >= 'a' && piece <= 'z')
339 piece = piece - 'a' + 'A'; // Capitalize piece for switch statement
340 switch (piece) {
341 case 'K':
342 generateKingMoves(boardArray,&moves,i,currentPlayer);
343 break;
344 case 'B':
345 generateBishopMoves(boardArray,&moves,i,currentPlayer);
346 break;
347 case 'R':
348 generateRookMoves(boardArray,&moves,i,currentPlayer);
349 break;
350 case 'Q':
351 generateQueenMoves(boardArray,&moves,i,currentPlayer);
352 break;
353 case 'P':
354 generatePawnMoves(boardArray,&moves,i,currentPlayer);
355 break;
356 case 'N':
357 generateKnightMoves(boardArray,&moves,i,currentPlayer);
358 break;
359 default:
360 break;
361 }
362 }
363 }
364 SafeFree(boardArray);
365 return moves;
366}

369/************************************************************************
370**
371** NAME: DoMove
372**
373** DESCRIPTION: Applies the move to the position.
374**
375** INPUTS: POSITION position : The old position
376** MOVE move : The move to apply to the position
377**
378** OUTPUTS: (POSITION) : The position that results from move
379**
380** CALLS: Some Board Hash Function
381** Some Board Unhash Function
382**
383*************************************************************************/
384
385POSITION DoMove (POSITION position, MOVE move)
386{
387 char *boardArray = unhashBoard(position);
388 int filef;
389 int rankf;
390 int filei;
391 int ranki;
392 char pieceMoved;
393 int sideLength = (int)sqrt(strlen(boardArray));
394 int i, length = strlen(boardArray), minx = sideLength, x;
395 //ANDing by 2^(6)-1 will give the last 6 bits
396 for (i = 0; i < length; i++) {
397 if (boardArray[i] != ' ') {
398 x = i % sideLength;
399 if (x < minx) {
400 minx = x;
401 }
402 }
403 }
404 filef = move & 63;
405 rankf = ((move >> 6) & 63) + minx - 1;
406 filei = (move >> 12) & 63;
407 ranki = ((move >> 18) & 63) + minx - 1;
408 pieceMoved = boardArray[(sideLength-filei)*sideLength+ranki];
409 boardArray[(sideLength-filei)*sideLength+ranki] = ' ';
410 boardArray[(sideLength-filef)*sideLength+rankf] = pieceMoved;
411 //printf("In DoMove: \n");
412 //printProperBoard(boardArray);
413 boardArray = flushBoard(boardArray);
414 //printf("After flushing the board back: \n");
415 //printProperBoard(boardArray);
416 boardArray = switchBoardSize(boardArray);
417 return hashBoard(boardArray,(getCurrTurn(position) == PLAYER1_TURN) ? PLAYER2_TURN : PLAYER1_TURN);
418}

421/************************************************************************
422**
423** NAME: Primitive
424**
425** DESCRIPTION: Returns the value of a position if it fulfills certain
426** 'primitive' constraints.
427**
428** Example: Tic-tac-toe - Last piece already placed
429**
430** Case Return Value
431** *********************************************************
432** Current player sees three in a row lose
433** Entire board filled tie
434** All other cases undecided
435**
436** INPUTS: POSITION position : The position to inspect.
437**
438** OUTPUTS: (VALUE) : one of
439** (win, lose, tie, undecided)
440**
441** CALLS: None
442**
443************************************************************************/
444
445VALUE Primitive (POSITION position)
446{
447 char *boardArray = unhashBoard(position);
448 int currentPlayer = getCurrTurn(position);
449 if (inCheck(boardArray,currentPlayer) && !canMove(boardArray,currentPlayer)) {
450 SafeFree(boardArray);
451 // The king is checked and can't move
452 return (gStandardGame) ? lose : win;
453 }
454 else if (!inCheck(boardArray,currentPlayer) && !canMove(boardArray,currentPlayer)) {
455 // King is not in check and can't move - Stalemate
456 SafeFree(boardArray);
457 return tie;
458 } else {
459 SafeFree(boardArray);
460 return undecided;
461 }
462}

467** NAME: maxInt PrintPosition
468**
469** DESCRIPTION: Prints the position in a pretty format, including the
470** prediction of the game's outcome.
471**
472** INPUTS: POSITION position : The position to pretty print.
473** STRING playersName : The name of the player.
474** BOOLEAN usersTurn : TRUE <==> it's a user's turn.
475**
476** CALLS: Unhash()
477** GetPrediction() : Returns the prediction of the game
478**

480
481void PrintPosition (POSITION position, STRING playersName, BOOLEAN usersTurn)
482{
483 char *boardArray;
484 int r, c, sideLength, x, y;
485 boardArray = unhashBoard(position);
486 sideLength = sqrt(strlen(boardArray));
487 int bi=0, iterations = 0;
488 int minx=sideLength, maxx=0, miny=sideLength, maxy=0;
489
490 /* Determine the rectangle the pieces reside in.
491 This is done so as not to print unnecessary empty
492 rows and columns. */
493 for(bi=0; bi<strlen(boardArray); bi++) {
494 if(boardArray[bi]!=' ') {
495 x = bi%sideLength;
496 y = bi/sideLength;
497 if(x<minx) minx=x;
498 if(x>maxx) maxx=x;
499 if(y<miny) miny=y;
500 if(y>maxy) maxy=y;
501 }
502 }
503 printf("\n");
504 for(r = miny-1; r <= maxy+1; r++) {
505 printf(" +");
506 for(c = minx-1; c <= maxx+1; c++) {
507 printf("---+");
508 }
509 printf("\n");
510 printf(" %d |", sideLength - r);
511 for(c = minx-1; c <= maxx+1; c++) {
512 printf(" %c |", boardArray[r*sideLength+c]);
513 }
514 printf("\n");
515 }
516 printf(" +");
517 for(c = minx-1; c <= maxx+1; c++) {
518 printf("---+");
519 }
520 printf("\n");
521 printf(" ");
522 iterations = (maxx + 1) - (minx - 1) + 1;
523 for(c = 0; c < iterations; c++) {
524 printf(" %c ", 97+c);
525 }
526 printf("\n");
527 printf("%s\n",GetPrediction(position,playersName,usersTurn));
528 printf("It is %s's turn (%s).\n",playersName,(usersTurn) ? "white/uppercase" : "black/lowercase");
529 SafeFree(boardArray);
530}

610/************************************************************************
611**
612** NAME: ValidTextInput
613**
614** DESCRIPTION: Rudimentary check to check if input is in the move form
615** you are expecting. Does not check if it is a valid move.
616** Only checks if it fits the move form.
617**
618** Reserved Input Characters - DO NOT USE THESE ONE CHARACTER
619** COMMANDS IN YOUR GAME
620** ?, s, u, r, h, a, c, q
621** However, something like a3
622** is okay.
623**
624** Example: Tic-tac-toe Move Format : Integer from 1 to 9
625** Only integers between 1 to 9 are accepted
626** regardless of board position.
627** Moves will be checked by the core.
628**
629** INPUTS: STRING input : The string input the user typed.
630**
631** OUTPUTS: BOOLEAN : TRUE if the input is a valid text input.
632**
633************************************************************************/
634
635BOOLEAN ValidTextInput (STRING input)
636{
637 if(strlen(input)!=4)
638 return FALSE;
639 else{
640 char c = input[0];
641 /* Make sure the rank and file are within the bounds
642 set by the current board size. */
643 if (c >= 'a' && c <= 'z') c = c - 'a' + 'A';
644 if(c < 'A' || c > 'Z') return FALSE;
645 c = input[2];
646 if (c >= 'a' && c <= 'z') c = c - 'a' + 'A';
647 if(c < 'A' || c > 'Z') return FALSE;
648 c = input[1];
649 if(c < '1' || c > '9') return FALSE;
650 c = input[3];
651 if(c < '1' || c > '9') return FALSE;
652 }
653 return TRUE;
654}
