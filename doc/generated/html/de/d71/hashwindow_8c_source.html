35/*
36 VARIABLES:
37 BOOLEAN gHashWindowInitialized : tells if the hash window is initialized
38 BOOLEAN gCurrentTierIsLoopy : tells if the current tier is loopy
39 TIER* gTierInHashWindow : an array that tells what tiers are in the window
40 TIERPOSITION* gMaxPosOffset : an array that tells the max_pos of that tier
41 int gNumTiersInHashWindow : the number of tiers in the hash window
42
43 Hash Windows are set up like this:
44 First of all, Tier -1 (kBadTier) is ALWAYS in every Hash Window, and is
45 ALWAYS the first tier in the list, and ALWAYS contains 0 positions.
46 Don't ask why; just know that it makes things easier. :)
47
48 Now, say we are setting a window for Tier 0. Calling TierChildren on it returns
49 (1, 2, 3, 0). First, Tier 0 is placed in the front of the list, and if the
50 list already contains 0, it is removed. So the result is (0, 1, 2, 3).
51
52 Thus, TierInHashWindow becomes = { -1(kBadTier), 0, 1, 2, 3 }.
53 gNumTierInHashWindow is 5, since there are 5 tiers in the hash window
54 (including kBadTier).
55
56 Now, say the MaxPositions of the tiers are as follows:
57 0: 20 Positions (0 through 19)
58 1: 30 Positions (0 through 29)
59 2: 15 Positions (0 through 14)
60 3: 35 Positions (0 through 34)
61 Then, MaxPosOffset becomes = { 0, 20, 50, 65, 100 }.
62 gNumberOfPositions is set to 100.
63
64 Now in this window, we know 0-19 is Tier 0, 20-49 is Tier 1, 50-64 is Tier 2,
65 and 65-99 is Tier 3. Subtract the appropriate offsets and you get the original
66 MaxPositions for the tiers.
67
68 w00t.
69 -Max
70 */

74// Called by "Unhash".
75void gUnhashToTierPosition(POSITION position, TIERPOSITION* tierposition,
76 TIER* tier) {
77 if (!gHashWindowInitialized) {
78 printf("ERROR: Hash Window is not initialized!\n");
79 ExitStageRight();
80 } if (position >= gNumberOfPositions) {
81 printf("ERROR: Hash Window function \"gUnhashToTierPosition\" called with\n"
82 " illegal POSITION: %llu\n"
83 "(Current Hash Window's range is from 0 to %llu)\n",
84 position, gNumberOfPositions-1);
85 ExitStageRight();
86 }
87 // since we know position is legal, this works:
88 int i;
89 for(i = 1; i < gNumTiersInHashWindow; i++) {
90 if (position < gMaxPosOffset[i]) { //in THIS tier
91 (*tierposition) = position - gMaxPosOffset[i-1];
92 (*tier) = gTierInHashWindow[i];
93 return;
94 }
95 }
96}

98// Call by "Hash".
99POSITION gHashToWindowPosition(TIERPOSITION tierposition, TIER tier) {
100 if (!gHashWindowInitialized) {
101 printf("ERROR: Hash Window is not initialized!\n");
102 ExitStageRight();
103 }
104 int i;
105 for(i = 1; i < gNumTiersInHashWindow; i++) {
106 if (gTierInHashWindow[i] == tier) {
107 if (tierposition > gMaxPosOffset[i]) {
108 printf("ERROR: Hash Window function \"gHashToWindowPosition\" called with\n"
109 "illegal TIERPOSITION: %llu\n"
110 "(Tier %llu's reported range is from 0 to %llu)\n",
111 tierposition, tier, gMaxPosOffset[i]-1);
112 ExitStageRight();
113 }
114 return tierposition + gMaxPosOffset[i-1];
115 }
116 }
117 // shouldn't be reached. So, error:
118 printf("ERROR: Hash Window function \"gHashToWindowPosition\" called with\n"
119 "illegal TIER: %llu\n"
120 "(Current Hash Window includes these tiers:", tier);
121 for (i = 1; i < gNumTiersInHashWindow; i++)
122 printf(" %llu", gTierInHashWindow[i]);
123 printf(")\n");
124 ExitStageRight();
125 return 0;
126}

128/* FOR THE CORE (SOLVER/GAMESMAN) TO CALL */
129void gInitializeHashWindow(TIER tier, BOOLEAN loadDB) {
130 if (gHashWindowInitialized) {
131 if (gTierInHashWindow[1] == tier) //same hash window ALREADY initialized, so just return
132 return;
133 // Free old stuff
134 if (gMaxPosOffset != NULL) SafeFree(gMaxPosOffset);
135 if (gTierInHashWindow != NULL) SafeFree(gTierInHashWindow);
136 if (gTierDBExists != NULL) SafeFree(gTierDBExists);
137 }
138 gHashWindowInitialized = TRUE;
139
140 // Start by seeing what children go here
141 TIERLIST *children, *ptr, *back;
142 ptr = gTierChildrenFunPtr(tier);
143
144 // First, we find the list's size for malloc'ing, and insert TIER in its place.
145 gCurrentTierIsLoopy = FALSE;
146 int childrenCount = 1;
147 ptr = CreateTierlistNode(tier, ptr);
148 back = ptr; children = ptr->next;
149 for (; children != NULL; children = children->next, back = back->next) {
150 if (children->tier == tier) {
151 back->next = children->next;
152 gCurrentTierIsLoopy = TRUE;
153 } else childrenCount++;
154 }
155 gNumTiersInHashWindow = childrenCount+1;
156 gTierInHashWindow = (TIER*) SafeMalloc (gNumTiersInHashWindow * sizeof(TIER));
157 gMaxPosOffset = (TIERPOSITION*) SafeMalloc (gNumTiersInHashWindow * sizeof(TIERPOSITION));
158 gTierDBExists = (BOOLEAN*) SafeMalloc (gNumTiersInHashWindow * sizeof(BOOLEAN));
159
160 // Now we go through the list a second time to set the arrays.
161 children = ptr;
162 gTierInHashWindow[0] = kBadTier;
163 gMaxPosOffset[0] = 0;
164 gTierDBExists[0] = 0;
165 int i;
166 for (i = 1; i < gNumTiersInHashWindow; children = children->next, i++) {
167 gTierInHashWindow[i] = children->tier;
168 gMaxPosOffset[i] = gNumberOfTierPositionsFunPtr(children->tier)+gMaxPosOffset[i-1];
169 gTierDBExists[i] = FALSE; // the true values are set by tierdb.c
170 }
171 // set gNumberOfPositions
172 gNumberOfPositions = gMaxPosOffset[gNumTiersInHashWindow-1];
173 FreeTierList(ptr);
174 // finally, load the databases to memory:
175 // if gDontLoadTierDB is true, we have non-solve playing, so don't load db
176 // however, if static evaluator is on, then load as much as you can anyway
177 // however2, if static evaluator is PERFECT, then never load it
178 if (loadDB && (!gDontLoadTierDB || (gOpponent == AgainstEvaluator)) && !gSEvalPerfect) {
179 CreateDatabases();
180 InitializeDatabases();
181 if(!LoadDatabase()) {
182 printf("ERROR: Couldn't load tierDBs for Tier %llu!\n"
183 "(Current Hash Window includes these tiers:", tier);
184 for (i = 1; i < gNumTiersInHashWindow; i++)
185 printf(" %llu", gTierInHashWindow[i]);
186 printf(")\n");
187 ExitStageRight();
188 }
189 }
190 // just a few helper variables for the solver
191 gCurrentTier = gTierInHashWindow[1];
192 gCurrentTierSize = gMaxPosOffset[1];
193}

195// FOR GAMEPLAY.C
196void gInitializeHashWindowToPosition(POSITION* position, BOOLEAN loadDB) {
197 TIERPOSITION tierpos; TIER tier;
198 gUnhashToTierPosition((*position), &tierpos, &tier);
199 gInitializeHashWindow(tier, loadDB);
200 (*position) = gHashToWindowPosition(tierpos, tier);
201}

203BOOLEAN gTierDBExistsForPosition(POSITION position) {
204 if (!gHashWindowInitialized || gSEvalPerfect)
205 return FALSE;
206 TIERPOSITION tierposition; TIER tier; int i;
207 gUnhashToTierPosition(position, &tierposition, &tier);
208 for(i = 1; i < gNumTiersInHashWindow; i++)
209 if (gTierInHashWindow[i] == tier) //get value for THIS tier
210 return gTierDBExists[i];
211 return FALSE; // This actually can't happen (invariant)
212}
