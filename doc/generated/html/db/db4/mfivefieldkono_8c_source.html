176MOVELIST *GenerateMoves(POSITION hash) {
177 FFK_Board *newboard = unhash(hash);
178 BOOLEAN turn = newboard->turn;
179 MOVELIST *moves = NULL;
180
182 int even_len = 12;
183 for (int i = 0; i < even_len; i++) {
184 if (i != 4 && i != 9) {
185 evaluateEven(i, i - 2, turn, &moves, newboard->even_component);
186 evaluateEven(i, i + 3, turn, &moves, newboard->even_component);
187 }
188 if (i != 2 && i != 7) {
189 evaluateEven(i, i + 2, turn, &moves, newboard->even_component);
190 evaluateEven(i, i - 3, turn, &moves, newboard->even_component);
191 }
192 }
193
194 int odd_len = 13;
195 for (int j = 0; j < odd_len; j++) {
196 if (j != 0 && j != 5 && j != 10) {
197 evaluateOdd(j, j + 2, turn, &moves, newboard->odd_component);
198 evaluateOdd(j, j - 3, turn, &moves, newboard->odd_component);
199 }
200 if (j != 2 && j != 7 && j != 12) {
201 evaluateOdd(j, j - 2, turn, &moves, newboard->odd_component);
202 evaluateOdd(j, j + 3, turn, &moves, newboard->odd_component);
203 }
204 }
205
206 return moves;
207}

210POSITION DoMove(POSITION hash, MOVE move) {
211 int oldPos, newPos;
212 BOOLEAN turn = getTurn(hash);
213 unhashMove(move, &oldPos, &newPos);
214 int piece_type = turn ? 2: 1; // o == 1 and x == 2 in base 3
215 POSITION original = piece_type * pow(3, oldPos);
216 POSITION new = piece_type * pow(3, newPos);
217 return withTurn(hash - original + new, (turn + 1) % 2);
218}

221POSITION GetCanonicalPosition(POSITION position) {
222 POSITION* symmetries = malloc(sizeof(POSITION)*8);
223 FFK_Board* board = unhash(position);
224 POSITION canonical = 0;
225
226 symmetries[0] = position; // identity
227 rotate(board);
228 symmetries[1] = hash(board); // r
229 rotate(board);
230 symmetries[2] = hash(board); // r^2
231 rotate(board);
232 symmetries[3] = hash(board); // r^3
233 rotate(board);
234 flip(board);
235 symmetries[4] = hash(board); // f
236 rotate(board);
237 symmetries[5] = hash(board); // fr (frfr ong deadass)
238 rotate(board);
239 symmetries[6] = hash(board); // fr^2
240 rotate(board);
241 symmetries[7] = hash(board); // fr^3
242
243 for (int i = 0; i < 8; i++) {
244 if (symmetries[i] > canonical) {
245 canonical = symmetries[i];
246 }
247 }
248
249 free(symmetries);
250 free(board);
251
252 return canonical;
253}

410POSITION hash(FFK_Board *board) {
411 POSITION total = 0;
412 int even_len = 12;
413 for (int i = 0; i < even_len; i++) {
414 total += convertChar(board->even_component[(even_len - 1) - i]) * pow(3, i);
415 }
416 int odd_len = 13;
417 for (int j = 0; j < odd_len; j++) {
418 total += convertChar(board->odd_component[(odd_len - 1) - j]) * pow(3, 12 + j);
419 }
420 // Returns base_3(concatenate(odd_component, even_component))
421 // with the MSB flipped to board->turn
422 return withTurn(total, board->turn);
423}

426FFK_Board* unhash(POSITION hash) {
427 FFK_Board* newBoard = (FFK_Board *) malloc(sizeof(FFK_Board));
428 newBoard->even_component = (char *) malloc(sizeof(char)*13);
429 newBoard->odd_component = (char *) malloc(sizeof(char)*14);
430 newBoard->turn = getTurn(hash);
431
432 int remain = -1;
433 int even_len = 12;
434 newBoard->even_component[even_len] = '\0';
435 for (int i = 0; i < even_len; i++) {
436 remain = hash % 3;
437 hash = floor(hash/3);
438 newBoard->even_component[(even_len - 1) - i] = convertInt(remain);
439 }
440
441 int odd_len = 13;
442 newBoard->odd_component[odd_len] = '\0';
443 for (int j = 0; j < odd_len; j++) {
444 remain = hash % 3;
445 hash = floor(hash/3);
446 newBoard->odd_component[(odd_len - 1) - j] = convertInt(remain);
447 }
448 return newBoard;
449}
