47BOOL checkThreeMoves(int a, int b, int c)
48{
49 return ((movesDone[a] != 0) && (movesDone[a] == movesDone[b]) && (movesDone[a] == movesDone[c]));

52BOOL checkForFinishCondition()
53{
54 int i;
55 for (i = 0; i < 9; i++) {
56 if (movesDone[i] == 0) {
57 break;
58 }
59 }
60 if (i == 9) {
61 return TRUE; // board is full
62 }
63 return (checkThreeMoves(0, 1, 2) ||
64 checkThreeMoves(3, 4, 5) ||
65 checkThreeMoves(6, 7, 8) ||
66 checkThreeMoves(0, 3, 6) ||
67 checkThreeMoves(1, 4, 7) ||
68 checkThreeMoves(2, 5, 8) ||
69 checkThreeMoves(0, 4, 8) ||
70 checkThreeMoves(6, 4, 2));
71}

74// called by Tcl Event loop to check for new data
75void TclTimerProc(ClientData clientData)
76{
77 if (!isPenLoaderStarted()) {
78 printLog("Timer stopped\n");
79 return;
80 }
81
82 // get the next stroke to analyze
83 Stroke stroke = getNextAvailableStroke();
84 Tcl_CreateTimerHandler(TCL_TIMER_MS, &TclTimerProc, NULL);
85 if (stroke.nSamples == 0) {
86 return;
87 }
88 if (gameFinished) {
89 printLog(" This game is already finished, skipping stroke\n");
90 return;
91 }
92
93
94 // if already have the board, determine and make the move
95 if (numBoardLines == 4) {
96 Line testLine = newLine(boardCenterX, boardCenterY, stroke.bounds.centerX, stroke.bounds.centerY);
97 int crossing = (doLinesIntersect(&testLine, boardLeft)) ? CROSS_LEFT : CROSS_NONE;
98 crossing |= (doLinesIntersect(&testLine, boardRight)) ? CROSS_RIGHT : CROSS_NONE;
99 crossing |= (doLinesIntersect(&testLine, boardTop)) ? CROSS_TOP : CROSS_NONE;
100 crossing |= (doLinesIntersect(&testLine, boardBottom)) ? CROSS_BOTTOM : CROSS_NONE;
101
102 int move = CrossToMove[crossing];
103 if (move == -1) {
104 printLog(" Invalid crossing value detected: %d, skipping\n", crossing);
105 return;
106 } else if (movesDone[move] != 0) {
107 printLog(" Skipping duplicate move: %d\n", move);
108 return;
109 }
110
111 movesDone[move] = currentPlayer;
112 currentPlayer = (currentPlayer % 2) + 1;
113 printLog(" Making move %d\n", move);
114 char moveCommand[30];
115 sprintf(moveCommand, "ReturnFromHumanMove %d", move);
116 Tcl_Eval(tclInterpreter, moveCommand);
117 if (checkForFinishCondition()) {
118 gameFinished = TRUE;
119 printLog(" Game finished\n");
120 }
121 return;
122 }
123
124
125 // otherwise, add the stroke to the board
126 if (stroke.type != STROKE_LINE) {
127 printLog(" Board isn't finished, skipping non-LINE stroke\n");
128 return;
129 }
130 boardLines[numBoardLines] = stroke.line;
131 numBoardLines++;
132 if (numBoardLines < 4) {
133 return;
134 }
135
136
137 // determine board lines (two pairs of non-intersecting lines)
138 int idx, match1b, match2a, match2b;
139 match1b = 0;
140 for (idx = 1; idx < 4; idx++) {
141 if (!doLinesIntersect(&boardLines[0], &boardLines[idx])) {
142 if (match1b != 0) {
143 // matched twice, so something is wrong
144 match1b = 0;
145 break;
146 }
147 match1b = idx;
148 match2a = (idx == 1) ? 2 : 1;
149 match2b = (idx == 3) ? 2 : 3;
150 }
151 }
152
153 // verify that matched lines intersect properly
154 if ((match1b == 0) ||
155 (!doLinesIntersect(&boardLines[match1b], &boardLines[match2a])) ||
156 (!doLinesIntersect(&boardLines[match1b], &boardLines[match2b])) ||
157 (doLinesIntersect(&boardLines[match2a], &boardLines[match2b]))) {
158 printLog(" Could not match board lines, trying again without first line\n");
159 boardLines[0] = boardLines[1];
160 boardLines[1] = boardLines[2];
161 boardLines[2] = boardLines[3];
162 numBoardLines = 3;
163 return;
164 }
165
166 // assign proper sides (within 90 degree range away from pure vertical or horizontal)
167 double angle = fabs(boardLines[0].angleDegrees);
168 BOOL isLeft, isTop;
169 int leftIdx, rightIdx, topIdx, bottomIdx;
170 if (angle >= 45 && angle < 135) {
171 // match 1 is vertical, match 2 is horizontal
172 isLeft = (boardLines[0].centerX < boardLines[match1b].centerX);
173 isTop = (boardLines[match2a].centerY < boardLines[match2b].centerY);
174 leftIdx = isLeft ? 0 : match1b;
175 rightIdx = isLeft ? match1b : 0;
176 topIdx = isTop ? match2a : match2b;
177 bottomIdx = isTop ? match2b : match2a;
178 } else {
179 // match 2 is vertical, match 1 is horizontal
180 isLeft = (boardLines[match2a].centerX < boardLines[match2b].centerX);
181 isTop = (boardLines[0].centerY < boardLines[match1b].centerY);
182 leftIdx = isLeft ? match2a : match2b;
183 rightIdx = isLeft ? match2b : match2a;
184 topIdx = isTop ? 0 : match1b;
185 bottomIdx = isTop ? match1b : 0;
186 }
187 boardLeft = &boardLines[leftIdx];
188 boardRight = &boardLines[rightIdx];
189 boardTop = &boardLines[topIdx];
190 boardBottom = &boardLines[bottomIdx];
191
192 // calculate center of the board (average of the 4 intersection points)
193 double iLeftTop = intersectionPoint(boardLeft, boardTop);
194 double iLeftBottom = intersectionPoint(boardLeft, boardBottom);
195 double iRightTop = intersectionPoint(boardRight, boardTop);
196 double iRightBottom = intersectionPoint(boardRight, boardBottom);
197 boardCenterX = projectedX(boardLeft, iLeftTop, TRUE);
198 boardCenterX += projectedX(boardLeft, iLeftBottom, TRUE);
199 boardCenterX += projectedX(boardRight, iRightTop, TRUE);
200 boardCenterX += projectedX(boardRight, iRightBottom, TRUE);
201 boardCenterY = projectedY(boardLeft, iLeftTop, TRUE);
202 boardCenterY += projectedY(boardLeft, iLeftBottom, TRUE);
203 boardCenterY += projectedY(boardRight, iRightTop, TRUE);
204 boardCenterY += projectedY(boardRight, iRightBottom, TRUE);
205 boardCenterX /= 4;
206 boardCenterY /= 4;
207 printLog(" Detected board strokes: Left %d, Right %d, Top %d, Bottom %d, with center (%.3f, %.3f)\n", \
208 leftIdx, rightIdx, topIdx, bottomIdx, boardCenterX, boardCenterY);
209}
