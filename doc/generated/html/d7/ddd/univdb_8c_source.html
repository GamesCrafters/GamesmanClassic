70 /* Decide how many slots the database will have initially.

72 */
73 slots = (gNumberOfPositions > MAX_INIT_SLOTS) ? MAX_INIT_SLOTS : gNumberOfPositions;
74
75 /* Create hash table for database */

155VALUE univdb_put_value (POSITION position, VALUE value) {
156
157 univdb_entry *entry;
158
159 /* Obtain entry from hash-table */
160 entry = univdb_lookup_entry(position);
161
162 /* If no entry in hash-table, create new one */
163 if (entry == NULL) {
164
165 /* Create new hash table entry for position */
166 entry = univdb_create_entry(position);
167
168 }
169
170 /* Set new flags to entry to include for updated value */
171 entry->flags = (entry->flags & ~VALUE_MASK) | (value & VALUE_MASK);
172
173 /* Return filtered value */
174 return (entry->flags & VALUE_MASK);
175
176}
