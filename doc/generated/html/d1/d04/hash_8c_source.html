170/* initializes hash tables and critical values */
171POSITION generic_hash_init(int boardsize, int *pieces_array, int (*fn)(int *), int player)
172{
173 int i = -1, k;
174 int newcntxt;
175 POSITION sofar,temp;
176
177 //if(cCon == NULL)
178 // ExitStageRightErrorString("Attempting to Initialize invalid hash context");
179
180 //if(cCon->init)
181 // freeHashContext(cCon);
182
183 //cCon->init = TRUE;
184
185 newcntxt = generic_hash_context_init();
186
187 currentContext = newcntxt; //context switch
188
189 cCon->numPieces = hash_countPieces(pieces_array);
190 cCon->gfn = fn;
191
192 cCon->pieces = (char*) SafeMalloc (sizeof(char) * cCon->numPieces);
193 cCon->mins = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
194 cCon->maxs = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
195 cCon->nums = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
196 cCon->thisCount = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
197 cCon->localMins = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
198 cCon->gpdStore = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
199 cCon->miniOffset = (POSITION*) SafeMalloc (sizeof(POSITION) * (cCon->numPieces+2));
200 cCon->miniIndices = (int*) SafeMalloc (sizeof(int) * (cCon->numPieces+2));
201
202 getPieceParams(pieces_array, cCon->pieces, cCon->mins,cCon->maxs);
203 for (i = 0; i < cCon->numPieces; i++) {
204 cCon->nums[i] = cCon->maxs[i] - cCon->mins[i] + 1;
205 }
206
207 temp = 0;
208 cCon->numCfgs = 1;
209 for (i = 0; i < cCon->numPieces; i++) {
210 if(cCon->numCfgs < temp)
211 ExitStageRightErrorString("Generic Hash: Number of Cfg's wrap on current POSITION type");
212 cCon->numCfgs *= cCon->nums[i];
213 temp = cCon->numCfgs;
214 }
215 cCon->pieceIndices = (int*) SafeMalloc (sizeof(int) * cCon->numCfgs);
216
217
218 nCr_init(boardsize);
219 //hash_combiCalc();
220 cCon->hashOffset = (POSITION*) SafeMalloc(sizeof(POSITION) * (cCon->usefulSpace+2));
221 cCon->offsetIndices = (int*) SafeMalloc(sizeof(int) * (cCon->usefulSpace+2));
222 for(i = 0; i < cCon->usefulSpace; i++) {
223 cCon->hashOffset[i] = -1;
224 cCon->offsetIndices[i] = -1;
225 }
226 cCon->hashOffset[0] = 0;
227 cCon->offsetIndices[0] = 0;
228 for (k = 1; k <= cCon->usefulSpace; k++) {
229 cCon->hashOffset[k] = combiCount(gPieceDist(k-1));
230 cCon->offsetIndices[k] = gpi(k-1);
231 }
232 cCon->hashOffset[k] = -1;
233 temp = 0;
234 sofar = 0;
235 //printf("\n");
236 for (i = 1; i <= cCon->usefulSpace; i++) {
237 //printf("block: %llu Tot:%llu\n",cCon->hashOffset[i],sofar);
238 //fflush(NULL);
239 sofar += cCon->hashOffset[i];
240 if(sofar<temp) {
241 ExitStageRightErrorString("To many positions to represent in current position format");
242 }
243 cCon->hashOffset[i] = sofar;
244 cCon->offsetIndices[i]++;
245 temp = sofar;
246 }
247 cCon->hashOffset[cCon->usefulSpace + 1] = -1;
248 cCon->maxPos = sofar;
249
250 cCon->player = player % 3; // ensures player is either 0, 1, or 2
251
252 if (cCon->player != 0)
253 return sofar;
254 else return sofar*2;
255}

257/* initializes pascal's triangle and piece configuration tables */
258void nCr_init(int boardsize)
259{
260 int i, j, k, sum, ctr;
261 int *temp = (int*) SafeMalloc (sizeof(int) * cCon->numPieces);
262 /*hash_NCR = */
263 cCon->NCR = (POSITION*) SafeMalloc(sizeof(POSITION) * (boardsize + 1) * (boardsize + 1));
264 /*hash_boardSize = */
265 cCon->boardSize = boardsize;
266
267 /*store the left and right wings of pascal's triangle*/
268 for(i = 0; i<= boardsize; i++) {
269 cCon->NCR[i*(boardsize+1)] = 1;
270 cCon->NCR[i*(boardsize+1) + i] = 1;
271 }
272
273 /*compute and store the middle of pascal's triangle*/
274 for(i = 1; i<= boardsize; i++) {
275 for(j = 1; j < i; j++) {
276 cCon->NCR[i*(boardsize+1) + j] = cCon->NCR[(i-1)*(boardsize+1) + j-1] + cCon->NCR[(i-1)*(boardsize+1) + j];
277 }
278 }
279
280 /* compute the gPieceDist[i][j]'s with a method similar to Horner's method of
281 evaluating a polynomial p using only deg(p) multiplications
282 */
283 ctr = 0;
284 for(i=0; i<cCon->numCfgs; i++) {
285 cCon->pieceIndices[i] = -1;
286 }
287
288 for (i = 0; i < cCon->numCfgs; i++) {
289 sum = 0;
290 k = i;
291 for (j = 0; j < cCon->numPieces; j++) {
292 temp[j] = cCon->mins[j] + (k % (cCon->nums[j]));
293 sum += temp[j];
294 k = k/(cCon->nums[j]);
295 }
296 if (sum == boardsize && validConfig(temp)) {
297 cCon->pieceIndices[ctr] = i;
298 ctr++;
299 }
300 }
301 cCon->usefulSpace = ctr;
302 SafeFree(temp);
303}

306/* memoizes combinations not used right now */
307void hash_combiCalc()
308{
309 int i,sums;
310 int *thPieces = (int *) SafeMalloc(sizeof(int) * cCon->numPieces);
311
312 for(i=0; i<cCon->numPieces; i++)
313 thPieces[i] = cCon->mins[i];
314 sums = 1;
315 for(i=0; i<cCon->numPieces; i++) {
316 if(cCon->nums[i] != 0)
317 sums *= cCon->nums[i];
318 }
319 cCon->combiArray = (POSITION *) SafeMalloc(sizeof(POSITION) * sums);
320 while(thPieces[cCon->numPieces-1] <= cCon->maxs[cCon->numPieces-1]) {
321
322 sums = 0;
323 for(i=0; i<cCon->numPieces; i++) {
324 sums = combiHash(sums,cCon->nums[i],thPieces[i]);
325 }
326 cCon->combiArray[sums] = combiCount(thPieces);
327 thPieces[0]++;
328 i=0;
329 while(i<(cCon->numPieces-1) && thPieces[i] > cCon->maxs[i]) {
330 thPieces[i]=cCon->mins[i];
331 i++;
332 thPieces[i]++;
333 }
334 }
335
336}

344/* hashes *board to a POSITION */
345POSITION generic_hash_hash(char* board, int player) {
346 int i, j;
347 POSITION temp, sum;
348 int boardSize = cCon->boardSize; /*hash_boardSize;*/
349
350 for (i = 0; i < cCon->numPieces; i++)
351 {
352 cCon->thisCount[i] = 0;
353 cCon->localMins[i] = cCon->mins[i];
354 }
355
356 for (i = 0; i < boardSize; i++)
357 {
358 for (j = 0; j < cCon->numPieces; j++) {
359 if (board[i] == cCon->pieces[j]) {
360 cCon->thisCount[j]++;
361 }
362 }
363 }
364 sum = 0;
365 for (i = cCon->numPieces-1; i >= 0; i--)
366 {
367 sum += (cCon->thisCount[i] - cCon->mins[i]);
368 if (i > 0) {
369 sum *= cCon->nums[i-1];
370 }
371 }
372 temp = cCon->hashOffset[searchIndices(sum)];
373 temp += hash_cruncher(board);
374 if (temp > cCon->maxPos) {
375 ExitStageRightErrorString("generic_hash encountered position larger than maxPos.");
376 }
377 if (cCon->player != 0) // using single-player boards, ignore "player"
378 return temp;
379 else return temp + (player-1)*(cCon->maxPos); //accomodates generic_hash_turn
380}

382//accomodates generic_hash_turn and symmetries
383POSITION generic_hash_hash_sym(char* board, int player, POSITION offset, struct symEntry* symIndex)
384{
385 int i, j;
386 POSITION temp;
387 int boardSize = cCon->boardSize; /*hash_boardSize;*/
388
389 if (symIndex == NULL)
390 ExitStageRightErrorString("Invalid symmetry");
391
392 for (i = 0; i < cCon->numPieces; i++)
393 {
394 cCon->thisCount[i] = 0;
395 cCon->localMins[i] = cCon->mins[i];
396 }
397
398 for (i = 0; i < boardSize; i++)
399 {
400 for (j = 0; j < cCon->numPieces; j++) {
401 if (board[symIndex->sym[i]] == cCon->pieces[j]) {
402 cCon->thisCount[j]++;
403 }
404 }
405 }
406 temp = offset + hash_cruncher_sym(board, symIndex);
407 if (cCon->player != 0) // using single-player boards, ignore "player"
408 return temp;
409 else return temp + (player-1)*(cCon->maxPos); //accomodates generic_hash_turn
410}

412/* helper func from generic_hash_unhash() computes lexicographic rank of *board
413 among boards with the same configuration argument *thiscount */
414POSITION hash_cruncher (char* board)
415{
416 POSITION sum = 0;
417 int i = 0, k = 0, max1 = 0;
418 int boardSize = cCon->boardSize;
419
420 for(; boardSize>1; boardSize--) {
421 i = 0;
422
423 while (board[boardSize - 1] != cCon->pieces[i])
424 i++;
425 for (k = 0; k < i; k++) {
426 max1 = 1;
427 if (cCon->localMins[k] > 1)
428 max1 = cCon->localMins[k];
429
430 if (cCon->thisCount[k] >= max1) {
431 cCon->thisCount[k]--;
432 sum += combiCount(cCon->thisCount);
433 cCon->thisCount[k]++;
434 }
435 }
436 cCon->thisCount[i]--;
437 cCon->localMins[i]--;
438 }
439
440 return sum;
441}

497/* unhashes hashed to a board */
498char* generic_hash_unhash(POSITION hashed, char* dest)
499{
500 POSITION offst;
501 int i, j, boardSize;
502 hashed %= cCon->maxPos; //accomodates generic_hash_turn
503
504 boardSize = cCon->boardSize;
505 j = searchOffset(hashed);
506 offst = cCon->hashOffset[j];
507 hashed -= offst;
508 for (i = 0; i < cCon->numPieces; i++) {
509 cCon->localMins[i] = cCon->mins[i];
510 cCon->thisCount[i] = gpd(cCon->offsetIndices[j + 1] - 1)[i];
511 }
512 hash_uncruncher(hashed, dest);
513 return dest;
514}

516/* helper func from generic_hash_hash() computes a board, given its lexicographic rank hashed
517 among boards with the same configuration argument *thiscount*/
518void hash_uncruncher (POSITION hashed, char *dest)
519{
520 int i = 0, j = 0;
521 int max1 = 0;
522 int boardSize = cCon->boardSize;
523
524 for(; boardSize>0; boardSize--) {
525 if (boardSize == 1) {
526 i = 0;
527 while (cCon->thisCount[i] == 0)
528 i++;
529 dest[0] = cCon->pieces[i];
530 } else {
531 cCon->miniOffset[0] = 0;
532 cCon->miniIndices[0] = 0;
533 j = 1;
534 for (i = 0; (i < cCon->numPieces) && (cCon->miniOffset[j-1] <= hashed); i++) {
535 max1 = 1;
536 if (cCon->localMins[i] > 1)
537 max1 = cCon->localMins[i];
538 if (cCon->thisCount[i] >= max1) {
539 cCon->thisCount[i]--;
540 cCon->miniOffset[j] = cCon->miniOffset[j-1] + combiCount(cCon->thisCount);
541 cCon->thisCount[i]++;
542 cCon->miniIndices[j] = i;
543 j++;
544 }
545 }
546 i = j-1;
547 cCon->thisCount[cCon->miniIndices[i]]--;
548 cCon->localMins[cCon->miniIndices[i]]--;
549 dest[boardSize-1] = cCon->pieces[cCon->miniIndices[i]];
550 hashed = hashed - cCon->miniOffset[i-1];
551 }
552
553 }
554
555}

567/************************************
568**
569** generic_hash_context_init()
570**
571** Generates a new uninitialized
572** hash context with a unique hash
573** context identifier
574**
575************************************/
576int generic_hash_context_init()
577{
578 // initializes the context hashtable if it isn't already
579 if (!hashtableInitialized)
580 initializeHashtable();
581
582 int i;
583 struct hashContext **temp;
584 struct hashContext *newHashC;
585
586 int myContext = hash_tot_context;
587 hash_tot_context++;
588
589 temp = (struct hashContext **) SafeMalloc (sizeof(struct hashContext*)*(hash_tot_context));
590
591 if(contextList != NULL) {
592 for(i=0; i<(hash_tot_context-1); i++) //copy the last contexts over
593 temp[i] = contextList[i];
594 SafeFree(contextList);
595 contextList = temp;
596 } else {
597 contextList = temp;
598 }
599
600 newHashC = (struct hashContext *) SafeMalloc (sizeof(struct hashContext));
601 newHashC->hashOffset = NULL;
602 newHashC->maxPos = 0;
603 newHashC->NCR = NULL;
604 newHashC->gpdStore = NULL;
605 newHashC->offsetIndices = NULL;
606 newHashC->pieceIndices = NULL;
607 newHashC->boardSize = 0;
608 newHashC->numCfgs = 0;
609 newHashC->usefulSpace = 0;
610 newHashC->numPieces = 0;
611 newHashC->pieces = NULL;
612 newHashC->nums = NULL;
613 newHashC->mins = NULL;
614 newHashC->maxs = NULL;
615 newHashC->thisCount = NULL;
616 newHashC->localMins = NULL;
617 newHashC->gfn = NULL;
618 newHashC->player = 0;
619 //newHashC->init = FALSE;
620 newHashC->contextNumber = -1;
621 contextList[hash_tot_context-1] = newHashC;
622
623 return myContext;
624}

782 the members of *tc */

784{
785 POSITION sum = 0, prod = 1, ind = 0,old=1,hold=0;
786 for (ind = 0; ind < cCon->numPieces - 1; ind++) {
787 sum += tc[ind];
788 hold = nCr(sum+tc[ind+1], sum);

792 old = prod;
793 }

881void hashtablePut(int context, int index, int old) {
882 MOVELIST* bucket; BOOLEAN found;
883 int contextBucket = context % HASHTABLE_BUCKETS;
884 if (old != -1) {
885 // first, remove old context!
886 int oldContextBucket = contextList[index]->contextNumber % HASHTABLE_BUCKETS;
887 bucket = generic_hash_hashtable[oldContextBucket];
888 MOVELIST* pre = NULL;
889 found = FALSE;
890 for (; bucket != NULL; bucket = bucket->next) {
891 if (bucket->move == index) { // remove from list
892 if (pre == NULL) { // first one on list!
893 generic_hash_hashtable[oldContextBucket] = bucket->next;
894 } else {
895 pre->next = bucket->next;
896 }
897 SafeFree(bucket);
898 found = TRUE;
899 break;
900 }
901 pre = bucket;
902 }
903 if (!found) // if we DIDN'T find old hash context... big problem
904 ExitStageRightErrorString("Congratulations! You BROKE Gamesman! YAY!!!!!\n");
905 }
906 // Now, we can place the context
907 bucket = generic_hash_hashtable[contextBucket];
908 found = FALSE;
909 // let's make sure it's not already in there...
910 for (; bucket != NULL; bucket = bucket->next) {
911 if (contextList[bucket->move]->contextNumber == context) {
912 found = TRUE;
913 break;
914 }
915 }
916 if (found) { // duplicate context, error out.
917 printf("ERROR: Assigning hash context number %d, but it already exists!\n", context);
918 exit(1);
919 }
920 // FINALLY, we can place the context in the hashtable
921 generic_hash_hashtable[contextBucket] =
922 CreateMovelistNode(index, generic_hash_hashtable[contextBucket]);
923}
924
925int hashtableGet(int context) {
926 MOVELIST* bucket = generic_hash_hashtable[context % HASHTABLE_BUCKETS];
927 int index;
928 // find the index and return it
929 for (; bucket != NULL; bucket = bucket->next) {
930 index = bucket->move;
931 if (contextList[index]->contextNumber == context) {
932 return index;
933 }
934 }
935 // if got here, error
936 return -1;
937}

1306POSITION generic_hash_canonicalPosition(POSITION pos) {
1307 char* board = (char*) SafeMalloc(sizeof(char) * cCon->boardSize);
1308 char* flippedboard = (char*) SafeMalloc(sizeof(char) * cCon->boardSize);
1309 struct symEntry* symIndex = symmetriesList->next;
1310 generic_hash_unhash(pos, board);
1311 generic_hash_unhash(pos, flippedboard);
1312 int player = generic_hash_turn(pos);
1313 int flippedplayer;
1314 if (player == 1)
1315 flippedplayer = 2;
1316 else
1317 flippedplayer = 1;
1318 POSITION tempPos, sum, offset, minPos;
1319 POSITION flippedsum, flippedoffset;
1320 int i, j, boardSize = cCon->boardSize;
1321
1322 for (i = 0; i < cCon->numPieces; i++)
1323 {
1324 cCon->thisCount[i] = 0;
1325 cCon->localMins[i] = cCon->mins[i];
1326 }
1327
1328 for (i = 0; i < boardSize; i++)
1329 {
1330 for (j = 0; j < cCon->numPieces; j++) {
1331 if (board[symIndex->sym[i]] == cCon->pieces[j]) {
1332 cCon->thisCount[j]++;
1333 }
1334 }
1335 }
1336 sum = 0;
1337 for (i = cCon->numPieces-1; i >= 0; i--)
1338 {
1339 sum += (cCon->thisCount[i] - cCon->mins[i]);
1340 if (i > 0) {
1341 sum *= cCon->nums[i-1];
1342 }
1343 }
1344 offset = cCon->hashOffset[searchIndices(sum)];
1345
1346 if (symmetriesList->flip == 1) {
1347 //for flipped:
1348 for (i = 0; i < cCon->boardSize; i++) {
1349 if (flippedboard[i] == cCon->pieces[0])
1350 flippedboard[i] = cCon->pieces[1];
1351 else if (flippedboard[i] == cCon->pieces[1])
1352 flippedboard[i] = cCon->pieces[0];
1353 }
1354
1355 for (i = 0; i < cCon->numPieces; i++) {
1356 cCon->thisCount[i] = 0;
1357 cCon->localMins[i] = cCon->mins[i];
1358 }
1359
1360 for (i = 0; i < boardSize; i++)
1361 {
1362 for (j = 0; j < cCon->numPieces; j++) {
1363 if (flippedboard[symIndex->sym[i]] == cCon->pieces[j]) {
1364 cCon->thisCount[j]++;
1365 }
1366 }
1367 }
1368 flippedsum = 0;

1483 y = y->next;
