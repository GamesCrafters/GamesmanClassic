72/* walk the game tree in a BFS (no graphs please) and generate files with the names
73 "./stages/1210_stage1_positions.txt"
74 to aid solving the stuff*/
75void WalkGameTree()
76{
77 POSITIONQUEUE *head, *tail, *stagetail;
78 BOOLEAN isPrimitive, beginNewStage;
79 MOVELIST *currentMoves, *currentMovesHead;
80 POSITION currentPos, childPos;
81
82 tail = NULL;
83 AddPositionToQueue(gInitialPosition, &tail);
84 head = tail;
85 stagetail = tail;
86 TotalStages = -1;
87
88 mkdir("stages", 0755);
89
90 printf("I am walking the game tree now.\n");
91
92 beginNewStage = TRUE;
93
94 UnMarkAllAsVisited();
95
96 do {
97 currentPos = head->position;
98
99 /*write the position to the first file;*/
100 if (beginNewStage) {
101 TotalStages++;
102 printf("walking stage %d\n", TotalStages);
103 newStageFile();
104 beginNewStage = FALSE;
105 }
106
107 fprintf(StageFile, POSITION_FORMAT "\n", currentPos);
108 fflush(StageFile);
109
110 currentMoves = currentMovesHead = GenerateMoves(currentPos);
111
112 //this must hold true since we are always considering legal positions
113 //they can only lead to valid positions
114 //and even if primitives might have more moves ahead we stop already
115 isPrimitive = (currentMoves == NULL || Primitive(currentPos) != undecided);
116
117 if (!isPrimitive) {
118 for(; currentMovesHead != NULL; currentMovesHead = currentMovesHead->next) {
119 childPos = DoMove(currentPos, currentMovesHead->move);
120 if (!Visited(childPos)) {
121 AddPositionToQueue(childPos, &tail);
122 MarkAsVisited(childPos);
123 }
124 }
125 }
126
127 if (head == stagetail) //we have finished this stage, rotating
128 {
129 fclose(StageFile);
130 stagetail = tail;
131 beginNewStage = TRUE;
132 }
133
134 FreeMoveList(currentMoves);
135
136 RemovePositionFromQueue(&head);
137
138 } while (head != NULL);
139}
