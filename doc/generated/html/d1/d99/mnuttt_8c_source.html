65 * POSITIONs that come out of here must encode the entire game state, including
66 * the board layout and whose turn it is. so if i move my x one space down, i
67 * return a position representing that and the fact that it's o's turn.
68 * MOVEs need to encode anything having to do with a player's move, but note
69 * that they do not (necessarily) encode who performs it.
70 *
71 * from studying mjoust.c, it appears that both formats are arbitrary and left
72 * entirely up to us. note, though, that the definition implies that
73 * maintaining state is futile - we're only defining the rulepacks.
74 * redefinition of MOVE:
75 * fx,fy,tx,ty - ie use different base combination as discussed at last
76 * meeting. this is for undo compatibility, and to not

281/************************************************************************
282**
283** NAME: GenerateMoves
284**
285** DESCRIPTION: Creates a linked list of every move that can be reached
286** from this position. Returns a pointer to the head of the
287** linked list.
288**
289** INPUTS: POSITION position : Current position for move
290** generation.
291**
292** OUTPUTS: (MOVELIST *) : A pointer to the first item of
293** the linked list of generated moves
294**
295** CALLS: MOVELIST *CreateMovelistNode();
296**
297************************************************************************/
298
299MOVELIST *GenerateMoves (POSITION position)
300{
301 MOVELIST *moves = NULL;
302 int i, j, k, di, dj;
303 char board[BOARD_SIZE];
304 int player = generic_hash_turn(position);
305 BOOLEAN canProcessDir = FALSE;
306
307 generic_hash_unhash (position, board);
308 /* Use CreateMovelistNode(move, next) to 'cons' together a linked list */
309 for (i = 0; i < BOARD_ROWS; i++) {
310 for (j = 0; j < BOARD_COLS; j++) {
311 if (((player==1) && (board[Index(i,j)] == PLAYER1_PIECE)) ||
312 ((player==2) && (board[Index(i,j)] == PLAYER2_PIECE)))
313 {
314 for (k = 0; k < NUM_OF_DIRS; k++) {
315 // for every possible movement as specified in dir_increments
316 // except 5 and the diagonals when !CAN_MOVE_DIAGONALLY
317 if (k == 4)
318 canProcessDir = FALSE;
319 else if ((k == 0) || (k == 2) || (k == 6) || (k == 8))
320 canProcessDir = CAN_MOVE_DIAGONALLY;
321 else
322 canProcessDir = TRUE;
323 if (canProcessDir) {
324 di = i + dir_increments[k][0];
325 dj = j + dir_increments[k][1];
326 if ((di >= 0) && (dj >= 0) &&
327 (di < BOARD_ROWS) && (dj < BOARD_COLS) &&
328 (board[Index(di,dj)] == EMPTY_PIECE)) {
329 moves = CreateMovelistNode(Hasher(i,j,k),moves);
330 }
331 }
332 }
333 }
334 }
335 }
336 return moves;
337}

340/************************************************************************
341**
342** NAME: DoMove
343**
344** DESCRIPTION: Applies the move to the position.
345**
346** INPUTS: POSITION position : The old position
347** MOVE move : The move to apply to the position
348**
349** OUTPUTS: (POSITION) : The position that results from move
350**
351** CALLS: Some Board Hash Function
352** Some Board Unhash Function
353**
354*************************************************************************/
355
356POSITION DoMove (POSITION position, MOVE move)
357{
358 char board[BOARD_SIZE];
359 int current_player = generic_hash_turn(position);
360 int move_position = Unhasher_Index(move);
361 int row = Row (move_position);
362 int col = Column (move_position);
363 int direction = Unhasher_Direction(move);
364 int new_position = Index(row + dir_increments[direction][0], \
365 col + dir_increments[direction][1]);
366
367 generic_hash_unhash (position, board);
368 board[new_position] = board[move_position];
369 board[move_position] = (char) EMPTY_PIECE;
370 return generic_hash_hash(board, (current_player == PLAYER1_TURN ? PLAYER2_TURN : PLAYER1_TURN));
371}

374/************************************************************************
375**
376** NAME: Primitive
377**
378** DESCRIPTION: Returns the value of a position if it fulfills certain
379** 'primitive' constraints.
380**
381** Example: Tic-tac-toe - Last piece already placed
382**
383** Case Return Value
384** *********************************************************
385** Current player sees three in a row lose
386** Entire board filled tie
387** All other cases undecided
388**
389** INPUTS: POSITION position : The position to inspect.
390**
391** OUTPUTS: (VALUE) : one of
392** (win, lose, tie, undecided)
393**
394** CALLS: None
395**
396************************************************************************/
397
398VALUE Primitive (POSITION position)
399{
400 /* if it's your turn but there's already a win, then you must have lost!
401 * can think about it this way because players are only allowed to move their
402 * own pieces.
403 */
404 char board[BOARD_SIZE];
405 int x,y,d;
406
407 generic_hash_unhash(position, board);
408
409 for (y = 0; y < BOARD_ROWS; y++) /*horizontal*/
410 for (x = 0; x < BOARD_COLS+1-NUM_TO_WIN; x++)
411 if (board[Index(y,x)] != EMPTY_PIECE) {
412 for (d = 1; d < NUM_TO_WIN; d++)
413 if (board[Index(y,x+d-1)] != board[Index(y,x+d)])
414 break;
415 if (d == NUM_TO_WIN) return (MISERE ? win : lose);
416 }
417
418 for (y = 0; y < BOARD_ROWS+1-NUM_TO_WIN; y++) /*vertical*/
419 for (x = 0; x < BOARD_COLS; x++)
420 if (board[Index(y,x)] != EMPTY_PIECE) {
421 for (d = 1; d < NUM_TO_WIN; d++)
422 if (board[Index(y+d-1,x)] != board[Index(y+d,x)])
423 break;
424 if (d == NUM_TO_WIN) return (MISERE ? win : lose);
425 }
426
427 for (y = 0; y < BOARD_ROWS+1-NUM_TO_WIN; y++) { /*diagonal*/
428 for (x = 0; x < BOARD_COLS+1-NUM_TO_WIN; x++) {
429 if (board[Index(y,x)] != EMPTY_PIECE) {
430 for (d = 1; d < NUM_TO_WIN; d++)
431 if (board[Index(y+d-1,x+d-1)] != board[Index(y+d,x+d)])
432 break;
433 if (d == NUM_TO_WIN) return (MISERE ? win : lose);
434 }
435
436 if (board[Index(y+NUM_TO_WIN-1,x)] != EMPTY_PIECE) {
437 for (d = 1; d < NUM_TO_WIN; d++)
438 if (board[Index(y+NUM_TO_WIN-d,x+d-1)] != board[Index(y+NUM_TO_WIN-d-1,x+d)])
439 break;
440 if (d == NUM_TO_WIN) return (MISERE ? win : lose);
441 }
442 }
443 }
444
445 return undecided;
446}

601/************************************************************************
602**
603** NAME: ValidTextInput
604**
605** DESCRIPTION: Rudimentary check to check if input is in the move form
606** you are expecting. Does not check if it is a valid move.
607** Only checks if it fits the move form.
608**
609** Reserved Input Characters - DO NOT USE THESE ONE CHARACTER
610** COMMANDS IN YOUR GAME
611** ?, s, u, r, h, a, c, q
612** However, something like a3
613** is okay.
614**
615** Example: Tic-tac-toe Move Format : Integer from 1 to 9
616** Only integers between 1 to 9 are accepted
617** regardless of board position.
618** Moves will be checked by the core.
619**
620** INPUTS: STRING input : The string input the user typed.
621**
622** OUTPUTS: BOOLEAN : TRUE if the input is a valid text input.
623**
624************************************************************************/
625
626BOOLEAN ValidTextInput (STRING input)
627{
628 char *dirstr;
629 int i;
630 if ((strlen(input) != 4) && (strlen(input) != 5))
631 return FALSE;
632 if (input[2] != ' ')
633 return FALSE;
634 if ((input[0] < ROW_START) || (input[0] > (ROW_START+BOARD_COLS)) ||
635 (input[1] < '1') || (input[1] > '1'+BOARD_ROWS))
636 return FALSE;
637 dirstr = input+3; /*dir points at the first character of the direction*/
638 for (i = 0; i < NUM_OF_DIRS; i++)
639 if (strcmp(dirstr, directions[i]) == 0)
640 return TRUE;
641 return FALSE;
642}
